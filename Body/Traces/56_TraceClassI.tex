\subsection{Tracing Motions}\label{sec:trace}

\begin{Quote}
The primary motivation stems from the desire to eliminate cross sectional warping 
modes by association with the element strains \(\bm{\gamma},\bm{\kappa}\). 

An outline:
\begin{itemize}
    \item Introduce the \emph{trace} abstraction, which formalizes the ambiguity in representing the Saint-Venant class through classical fields \(\bm{u},\bm{\theta}\). 
    \item Investigate the special class of traces that exhibit Type 3 locality... they can be exactly represented with \emph{only} 6-DOF.
    \item Elaborate the effect of the trace on the boundary conditions that the beam model is capable of representing. 
\end{itemize}
\end{Quote}

% The eight warping shapes of \eqref{eq:def-eight-shapes} provide a complete basis 
% for representing particular solutions to Saint-Venant's continuum problems. 
% Any section-rigid pattern $\bm{v} + \bm{\omega} \times \bm{r}$ may be freely transferred between these components without changing the continuum displacement field. 


The ambiguity inherent in the representation \eqref{eq:trace-lift-linear}, is investigated systematically through the abstraction of a \emph{trace}. 
A trace $\mathcal{T}$ identifies 
a rule for associating one-dimensional fields to a given the continuum displacement \(\hat{\bm u}(x,\bm{r})\):
\begin{SaveEquation}{eq:trace-map}
\mathcal{T}: \hat{\bm{u}}(x, \bm{r}) \;\longmapsto\; 
\Big(\bm{u}(x),\, \bm{\theta}(x)
,\, \bm{\alpha}(x)
\Big)
% =:\bm{z}(x)
% \label{eq:trace-def}
\end{SaveEquation}
% such that the decomposition
% \begin{equation}
% \hat{\bm{u}}_{\rm P}(x, \bm{r}) = \bm{u}(x) + \bm{\theta}(x) \times \bm{r} 
% + \bm{\Phi}(\bm{r})\,\bm{\alpha}(x)
% \label{eq:trace-decomp}
% \end{equation}
% holds identically. 

There are infinitely many traces that may exactly represent a given solution \(\hat{\bm u}_{\rm P}\) to a Saint-Venant problem, and each imparts a distinct physical meaning to boundary conditions like \(\bm{\theta}_{\mathcal{T}}(x_0) = \mathbf{0}\) in a finite element simulation. 
It is natural to restrict attention to the finite dimensional space of traces which exhibit the following properties:
\begin{enumerate}
    \item Strain measures \(\bm{e}\) generated under Saint-Venant loading are at most linear in the axial coordinate $\reflenx$. 
    \item Any Saint-Venant strain field is exactly reconstructed from only the plane section strains \(\bm{\gamma}_{\mathcal{T}},\bm{\kappa}_{\mathcal{T}}\).
    \item Continuum plane-section deformations are preserved, in the sense that section‑rigid fields are mapped to pure beam rigid motions.
\end{enumerate}
% Properties (1)–(2) are enforced by the strain–affine and strain–exact conditions (D1), together with homogeneity of the resultant stiffness (C1). 
% Property (3) is enforced by the plane‑section condition (P1). 
% These restrictions motivate a sequence of definitions with corresponding propositions which eventually reduce traces of interest to an 18 dimensional space. Among these free parameters emerge the classical shear correction coefficients.

\begin{definition}[Axial Traces]
\label{def:sv-trace}
An axial \emph{trace} 
on a continuum motion \(\hat{\bm u}\) 
is a linear map
\[
\mathcal{T} :\CauchySpace\to C^1(I;\mathbb{R}^{6+n_w}),
\qquad
\hat{\bm{u}}(x,\bm{r}) \mapsto 
\bm{z}(x) =
\begin{pmatrix}
\bm{u}(x),&
\bm{\theta}(x),&
\bm{\alpha}(x)
\end{pmatrix},
\]
of the form
$$[\mathcal{T}\hat{\bm{u}}](x) = \mathcal{T}_x\bigl[\hat{\bm{u}}(x, \cdot)\bigr]$$
where $\{\mathcal{T}_x\}_{x \in I}$ is a family of bounded linear functionals on section displacements, \(\CauchySpace\) is the space of admissible motions \(\hat{\bm u}\), 
$I \subset \mathbb{R}$ is the axial interval, and the image is interpreted 
as a pair of centerline fields
\[
\bm{u} : I \to \mathbb{R}^3,
\qquad
\bm{\theta} : I \to \mathbb{R}^3.
\]
A trace $\mathcal{T}$ is linear if, for any 
$\hat{\bm{u}}_1,\hat{\bm{u}}_2 \in \CauchySpace$ and 
$\alpha,\beta \in \mathbb{R}$,
\[
\mathcal{T}[\alpha \hat{\bm{u}}_1 + \beta \hat{\bm{u}}_2]
=
\alpha\,\mathcal{T}[\hat{\bm{u}}_1] + \beta\,\mathcal{T}[\hat{\bm{u}}_2].
\]
A trace is prismatic if $\mathcal{T}_x = \mathcal{T}_0$ for all $x$. 
\end{definition}

\label{def:trace-lifting}
Let $\mathcal Z\coloneq C^1(I;\mathbb R^6)$ with $\bm z=(\bm u,\bm\theta)$, and

\[

\;\coloneq\;

\end{equation}
\end{definition}

\subsection{Saint-Venant Motions}

\begin{definition}[Saint~Venant Family]
\label{def:sv-family}
Let $\IesanMotions\subset \CauchySpace$ denote the (six-parameter) linear space of
Saint~Venant \emph{particular} displacement fields
\[
  \IesanVector\in\mathbb R^6 \;\longmapsto\; \hat{\bm u}_{\rm P}(\IesanVector)\in\IesanMotions,
\]
where $\IesanVector$ are the
Saint~Venant load parameters used in \eqref{eq:def-iesan-vector}
and $\hat{\bm u}_{\rm P}(\IesanVector)$ is given by \eqref{eq:iesan-general-solution}.

Let $\RigidMotions\subset\CauchySpace$ be the $6$-dimensional rigid-body subspace
\[
  \bm q\in\mathbb R^6 \;\longmapsto\; \hat{\bm u}_{\rm R}(\bm q)
  =\bm q_u + \bm q_\theta\times(x\FrameBasis+\bm{r}).
\]
The twelve-dimensional vector space of 
Saint-Venant displacement fields solving \eqref{eq:iesan-problem-1} and \eqref{eq:iesan-problem-2} is:
\[
  \mathcal U_{\rm SV}\coloneq \IesanMotions\oplus \RigidMotions
  =\{\hat{\bm u}_{\rm P}(\IesanVector)+\hat{\bm u}_{\rm R}(\bm q)\}.
\]
\end{definition}


\begin{definition}[SV-Compatible]
\label{def:sv-compatible-lifting}
A lifting $\mathcal T^{-1}$ (Def.~\ref{def:trace-lifting}) is called \emph{SV-compatible}
with $\mathcal T$ if, for every Saint~Venant motion
$\hat{\bm u}=\hat{\bm u}_{\rm P}(\IesanVector)+\hat{\bm u}_{\rm R}(\bm q)\in\mathcal U_{\rm SV}$,
there exist $\bm z\in\mathcal Z$ and $\bm\alpha\in\mathcal A_{n_w}$ such that
\[
\hat{\bm u} \equiv \mathcal T^{-1}[\bm z,\bm\alpha]
\quad\text{(up to a rigid field, if you only care about strain equivalence),}
\]
and $\bm z$ agrees with the trace:
\[
\big(\bm{z}(x),\bm{\alpha}(x)\big)=\mathcal T[\hat{\bm u}](x)\qquad \forall x\in I.
\]

A warping operator $\bm\Phi:\mathbb R^{n_w}\to\mathcal W(\Section)$ is \emph{SV-compatible}
if there exists a linear map
\[
\bm A:\mathbb R^6\to C^1(I;\mathbb R^{n_w}),\qquad
\IesanVector\mapsto \bm\alpha_{\rm SV}(\cdot;\IesanVector),
\]
such that for every Saint~Venant particular field $\hat{\bm u}_{\rm P}(\IesanVector)$,
\[
\hat{\bm u}_{\rm P}(\IesanVector)\;\equiv\; \mathcal T^{-1}\big[\bm z_{\rm SV}(\IesanVector),\bm\alpha_{\rm SV}(\IesanVector)\big]
\quad\text{(modulo a rigid field, or strain-equivalence).}
\]
\end{definition}

\begin{definition}[Saint~Venant Equivalence]
\label{def:sv-equivalence}
Two traces $\mathcal T_1,\mathcal T_2$ are \emph{SV-equivalent} if
$\mathcal T_1[\hat{\bm u}]=\mathcal T_2[\hat{\bm u}]$ for all $\hat{\bm u}\in\mathcal U_{\rm SV}$.
All “trace matrix” objects below characterize only the SV-equivalence class.
\end{definition}

\subsubsection{Class I: Kinematic Consistency}

\subparagraph{T1}
It is useful to restrict our attention to traces generating strains \(\bm{e}(x)\) which are at most linear in $x$ for any problem in the Saint-Venant class.
This ensures that the strains can be exactly represented by typical beam finite elements.

\begin{definition}[Class I]\label{def:class-1} % def:e1
A trace \(\mathcal{T}\) is called a \textbf{Class I} trace if:
\begin{enumerate}[label=(\roman*)]
    \item For all \(\hat{\bm u} \in \IesanMotions\oplus \RigidMotions\) the following relations are furnished by $x$-independent linear operators \(\TraceEvolveEE\),\(\TraceDeDpOne\), and \(\TraceStiffNM\in\mathrm{GL}(6)\):
    \begin{subequations}
    \begin{align}
        \bm{e}'(x) &=: \TraceEvolveEE\,\bm{e}(0) 
        \label{eq:req-t1} 
        \\ 
        \partial_{\IesanVector}\bm{e}'(x) &=: \TraceDeDpOne 
        \label{eq:req-d1}
        \\ 
        \partial_{\bm{e}} \snm(x) &=: \TraceStiffNM 
        \label{eq:req-c1}
    \end{align}
    \label{eq:class-i-a}
    \end{subequations}

    \item For all section-rigid fields \(\hat{\bm u} =\bm{v}(x)+\bm{\omega}(x)\times\bm{r} \in \PlaneMotions\), the trace fields \(\bm{u},\bm{\theta} = \mathcal{T} [\hat{\bm u}]\) are:
    \begin{subequations}
    \begin{align}
        \bm{u}(x) &= \hat{\bm u}(x,\mathbf{0}) \\
        \bm{\theta}(x) &= \bm{\omega}(x)
    \end{align}
    \label{eq:req-p1}
    \end{subequations}
\end{enumerate}
\end{definition}
%
The plane section property \eqref{eq:req-p1} ensures that for the few continuum solutions where plane sections \emph{do} remain plane, the rotation field \(\bm{\theta}\) becomes exactly the physical cross section rotation.
For example, in the simple \(\mathrm{P}_{\rm I}\) case of a cantilever loaded by a moment, \eqref{eq:req-p1} limits the traces under consideration to those which would not introduce spurious spring-like behavior at the root through the boundary condition \(\bm{\theta}(0) = \mathbf{0}\). 
In problems where warping does arise, this spring phenomenon becomes inescapable. 


Strains \(\bm{e}\) generated from a trace with the property \eqref{eq:class-i-a} can be written in the form:
\begin{equation}
\bm{e}_{\rm z}(x) 
% = \TraceDeDpRoot(\mathbf{1} + x\TraceEvolveEP)\bm{p}
=(\mathbf{1} + x\TraceEvolveEE)\TraceDeDpRoot\bm{p}
\label{eq:def-T0}
\end{equation}
\begin{remark}
A trace is not uniquely determined by its action on any finite-dimensional subspace of $\mathcal{U}(\Omega)$. 
In particular, infinitely many distinct traces may agree on the twelve-dimensional Saint~Venant subspace $\bar{\mathrm{P}} \subset \mathcal{U}(\Omega)$. Nevertheless, for beam analysis, the behavior on $\bar{\mathrm{P}}$ is precisely what matters: it determines the stiffness matrix, the strain measures, and the physical meaning of boundary conditions. 
We therefore characterize traces by their restriction to $\bar{\mathrm{P}}$, with the understanding that this characterization defines an equivalence class of section operators rather than a unique trace.
\end{remark}
\begin{proposition}\label{prop:trace-matrix}
The equivalence class (\cref{def:sv-equivalence}) of traces with the properties \eqref{eq:class-i-a} is uniquely represented in the Saint~Venant class by a matrix \(\TraceDpDeRoot \in\mathrm{GL}(6)\), and 
\begin{subequations}
    \begin{align} 
        \TraceEvolveEE &= \TraceDeDpRoot \IesanEvolveSP \TraceDpDeRoot \\
        \TraceDeDpOne &= \TraceDeDpRoot \IesanEvolveSP \label{eq:res-d1} \\
        \TraceStiffNM &= \IesanStiff \TraceDpDeRoot \label{eq:def-CT}
    \end{align}
\end{subequations}
where \(\IesanEvolveSP\) is the evolution operator given by \eqref{eq:A-matrix-neg}.
\end{proposition}

\begin{proof}
The plane section property constrains how $\TraceDeDpRoot$ acts on the section-rigid part of each Saint Venant mode.

\textit{Extension} ($a_\varepsilon$): The section-rigid displacement is 
$x\,a_\varepsilon\FrameBasis$, giving $\bm{\theta} = \mathbf{0}$ and 
$\bm{u} = x\,a_\varepsilon\FrameBasis$. The strains are 
$\epsilon = a_\varepsilon$ with all other components zero. 
This fixes column 1 of $\TraceDeDpRoot$ to $(1, \mathbf{0}, 0, \mathbf{0})^\top$.

\textit{Bending} ($\bm{a}_\Omega$): The section-rigid rotation is 
$\bm{\theta} = -x\,\FrameBasis \times \bm{a}_\Omega$, giving curvature 
$\bm{\kappa}_\perp = -\FrameBasis \times \bm{a}_\Omega$. 

The shear strain 
$\bm{\gamma}_\perp$ receives no section-rigid contribution. 
This fixes columns 2--3: entries $(1, 2\text{--}3) = \CenterTrace^\top$, 
$(2\text{--}3, 2\text{--}3) = \mathbf{0}$, $(4, 2\text{--}3) = 0$, and 
$(5\text{--}6, 2\text{--}3) = -\FrameBasis^\times$.

\textit{Torsion} ($a_\vartheta$): The section-rigid rotation is 
$\bm{\theta} = x\,a_\vartheta\FrameBasis$, giving twist $\varkappa = a_\vartheta$. 
This fixes column 4 entries $(1, 4) = 0$, $(4, 4) = 1$, $(5\text{--}6, 4) = \mathbf{0}$.
The shear--torsion coupling $\TraceFT$ and curvature--torsion coupling $\TraceBT$
arise from warping and remain free.

\textit{Flexure} ($\bm{b}_{\Section}$): The section-rigid part contributes to 
$\epsilon$ and $\bm{\kappa}_\perp$ through the $x$-dependent terms, 
but these appear in $\TraceDeDpOne = \TraceDeDpRoot\IesanEvolveSP$, not $\TraceDeDpRoot$. 
All entries in columns 5--6 arise from warping contributions and remain free 
parameters: $\TraceNF$ (axial--flexure), $\TraceFF$ (shear--flexure), 
$\TraceTF$ (twist--flexure), and $\TraceMF$ (curvature--flexure).

Counting the free parameters: $\CenterTrace$ (2) + $\TraceNF$ (2) + 
$\TraceFT$ (2) + $\TraceFF$ (4) + $\TraceTF$ (2) + $\TraceMF$ (4) + $\TraceBT$ (2) = 18.
\end{proof}

\paragraph{Implementation.}
In view of the Saint~Venant strains \eqref{eq:iesan-point-strain}, the warping shapes \(\bm\Phi\) must contain:
\[
% \bm{\Phi}=
\Big\{
  \One,
\ -\bm{r}^{\times},
\ \bm{W}_{(a)},
\ \bm{\Pi}_{(b)},
\ \FrameBasis\,\varphi,
\ \FrameBasis\otimes\bm{\psi}
\Big\}.
\]
Note that because this contains the plane-rigid span \(\PlaneMotions\), the primary trace fields \(\bm{u},\bm{\theta}\)
will not generally represent the full plane-rigid content, which becomes offloaded onto the warping field \(\bm\alpha\) through a linear operator \(\RigidShift\) determined by the trace. 
Upon elimination of the warping fields \(\bm\alpha\) the material strains are 
\begin{SaveEquation}{eq:eps-B-e}
\StrainVector(\reflenx,\bm{r}) = \Big(\RigidStrainMatrix (\mathbf{1} + \RigidShift) + \WarpStrainMatrix \ShapeShift\Big) \,\bm{e}(\reflenx)
\end{SaveEquation}
with \(\bm{e}(x) = (\bm\gamma(x),\,\bm{\kappa}(x))\) and:
\begin{equation*}
\begin{aligned}
\RigidStrainMatrix(\bm{r}) &:= [\mathbf{1}, \;-\bm{r}^{\times}]_{3\times6} \\
\WarpStrainMatrix(\bm{r}) &:= [\nabla\WarpTwistIesan,\; \FrameBasis_{\beta}\otimes\nabla_{\beta}\bm{\WarpShearIesan} + \bm{\Pi}_{(b)}]_{3\times 3}
\end{aligned}
\end{equation*}
In view of \eqref{eq:def-T0}, the strain at \(\reflenx=0\) is:
\begin{equation}
\bm{\varepsilon}(0,\bm{r}) = \Big(\RigidStrainMatrix (\bm{r})(\mathbf{1} + \RigidShift) + \WarpStrainMatrix(\bm{r})\ShapeShift\Big) \,\TraceDeDpRoot \bm{p}
\label{eq:trace-strain}
\end{equation}
Equating the \(\reflenx\)-constant strains in \eqref{eq:iesan-point-strain} with \eqref{eq:trace-strain} and exploiting the block structure of \(\TraceDpDeRoot\) \eqref{eq:T0-inv} furnishes:
\begin{equation}
\RigidShift = \begin{bmatrix}
\FrameBasis\otimes\ShiftN - \oneS & \ShiftScaleN\FrameBasis\otimes\FrameBasis %+ \FrameBasis\otimes(\CenterTrace\times\FrameBasis) 
\\
-\FrameBasis^{\times}\ShiftA + %\FrameBasis\otimes\ShiftNK + 
\FrameBasis\otimes\ShiftB^{\top}\CenterShearLocation & (\ShiftScaleT + \ShiftK\cdot\CenterShearLocation - 1)\FrameBasis\otimes\FrameBasis + (\ShiftF\times\FrameBasis)\otimes\FrameBasis
\end{bmatrix}_{6\times6}
\label{eq:class-ii-Le}
\end{equation}
and
\begin{equation}
\ShapeShift = 
\begin{bmatrix}
(\ShiftB^{\top}\CenterShearLocation + \ShiftT)^{\top} & (\CenterShearLocation\cdot\ShiftK + \ShiftScaleT)\FrameBasis^{\top} \\
\ShiftB & \ShiftK\otimes\FrameBasis
\end{bmatrix}_{3\times6}
\label{eq:class-i-La}
\end{equation}

\subsubsection{Summary.}
The Class I traces represent a family of beam theories parameterized by 18 constants which are all capable of exactly representing Saint~Venant solutions through Type 3 beam finite elements.

The lifting \eqref{eq:trace-lift-linear} is the common kinematic interface for both regimes:
(i) in Type~1 beam theories, $\bm\alpha(x)$ is retained as an independent field and a section model may be posed as
$(\bm\gamma,\bm\kappa,\bm\alpha,\bm\alpha')\mapsto(\bm n,\bm m,\bm w,\bm v)$; (ii) in Type~3 theories, $\bm\alpha,\bm\alpha'$
are eliminated and reconstructed internally from $\bm{e}=(\bm\gamma,\bm\kappa)$ through the Class~I trace data, after which the same
generalized section operator can be called and its output condensed back to a $6$-vector consistent with the element equilibrium operator.

In a Type 3 finite element simulation, the trace matrix \(\TraceDpDeRoot\) is incorporated into the section response through the elimination operators \(\RigidShift\) \eqref{eq:class-ii-Le} and \(\ShapeShift\) \eqref{eq:class-i-La}.
This determines:
\begin{itemize}
    \item The scaling of the warping shapes \(\bm\Phi\),
    \item How much of the plane-rigid content is carried by \(\bm{u},\bm{\theta}\)
\end{itemize}
Under all loadings in the Saint~Venant class \(\IesanMotions\) the element response exactly represents the continuum motion \(\hat{\bm u}\) through the lifting \eqref{eq:trace-lift-linear}.
