\begin{ambox}[State determination of the \ForceFrame{} element.]{box:MF}

Given target deformations \(\bm{v}\), find \(\bm{q}\):
\begin{enumerate}
    \item Initialize \(\dd \bm{q} = \mathbf{0}\).
    \item Perform \textbf{Update B}.
    \item While \(\|\residv\| > \epsilon\) repeat \textbf{Update A}.
    \item Compute \(\mathbf{K}_u\) \eqref{eq:mf-tangent}.
\end{enumerate}
where:
\begin{itemize}
    \item \textbf{Update A.} Given \(\residv\), update basic forces \(\bm{q}\) 
    \begin{enumerate}
      \item Set \(\dd \tilde{\bm{q}} = −\mathbf{F}_q^{-1} \tilde{\bm{g}}_{\rm v}\) and \(\eta=1\).
      \item Line search: While \(\phi(\eta) > \epsilon\):
      \begin{enumerate}
          \item Perform \textbf{Update B} with \(\dd \bm{q} = \eta\, \dd\tilde{\bm{q}}\) and evaluate \(\phi(\eta) = \|\residv\|\).
          \item Update \(\eta \leftarrow \tfrac{1}{2}\eta\). 
      \end{enumerate}
      \item Update \(\bm{q} \leftarrow \eta\,\dd\tilde{\bm{q}}\).
    \end{enumerate}

    \item \textbf{Update B.} Given \(\bm{v}\), \(\dd \bm{q}\) and a state \(\mathcal{N}_i = \{\bm{q}, \bm{e}_{\ell}\}\), advance \(\bm{e}\) and compute \(\tilde{\bm{g}}_{\rm v}\), \(\{\resids\}\) and \(\mathbf{F}_q\).
    \begin{enumerate}
    % \item Compute section deformation increments from force corrections: \(\delta \bm{e} = \mathbf{K}_s^{-1} (\ForceInterp \delta \bm{q})\)
    \item Section deformation update: \(\bm{e} \leftarrow \bm{e} + \dd \bm{e}\) with \(\dd \bm{e} = \mathbf{K}_s^{-1} (\ForceInterp \dd \bm{q})\)
    \item Perform section state determination (Box~\ref{box:impl}), get resultants \(\bm{s}_e\)
    \item Interpolate unbalanced forces \(\resids = \ForceInterp  \bm{q} + \bm{s}_p − \bm{s}_e\)
    \item Apply second deformation correction: \(\bm{e} \leftarrow \bm{e} + \mathbf{K}_s^{-1} \resids\)
    \item Integrate compatibility residual \eqref{eq:def-mv-flex}: \(\tilde{\bm{g}}_{\rm v} = −\bm{v} + \int  \ForceInterp^{\top} \bm{e} \,\dd x\) 
    \end{enumerate}
\end{itemize}
