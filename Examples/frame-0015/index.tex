\subparagraph{Context.}
This example demonstrates consistency of the moderate strain model (\cref{sec:std-section}) with the analytic solution \eqref{eq:0015-solution} derived by \cite{trahair2005nonlinear} for nonlinear torsion.

Similar problems have been considered by 
\cite{hsiao2000corotational,battini2002corotational,jonker2021threedimensional} \cite{rong2020geometrically} \cite{zhang2011formulation}.

\subparagraph{Setup.}
A shaft is subjected to a concentrated torque $\TwistRxn$ acting at $\reflenx=L$. 
\begin{Details}
Rotation is prevented at the left support, $\reflenx=0$. 
The boundary condition \(N=0\) requires:
$$
\int_A \AxialStress \dA
=\AxialRigidity\epsilon + \frac{1}{2} \int_A E r^2 \dA \,\Twist^2 =0
\qquad\text{so}\qquad 
\epsilon = -\frac{1}{2\AxialRigidity} \EIo \Twist^2
$$
The axial stress is:
$$
\AxialStress=\frac{1}{2} E\left(r^2-\frac{1}{A} \int_A r^2 \dA \right)\Twist^2
$$
\end{Details}
The differential equation of torsion equilibrium becomes \citep{trahair2005nonlinear}:
%
\begin{equation}
\TwistRigidity \vartheta' - \EJw \vartheta''' + \tfrac{1}{2} E I_n\left(\vartheta'\right)^3 = \TwistRxn
\tag{$\dagger$}
\end{equation}
%
where $\TwistRxn$ is the applied torque along the member, $\TwistRigidity$ is the uniform torsion constant \eqref{eq:def-j-origin}, $\EJw$ is the warping torsion constant \eqref{eq:Jw}, and $EI_n$ is the  nonlinear Wagner section constant;
$$
\begin{aligned}
I_n= & {\left[I_{p p}-4\left(y_0 \beta_x I_x+x_0 \beta_y I_y\right)+2\left(x_0^2-y_0^2\right)\left(I_x-I_y\right)+\left(x_0^2+y_0^2\right)^2 A\right] } \\
& -\left(I_p / A+x_0^2+y_0^2\right)^2 A-\beta_x^2 I_x-\beta_y^2 I_y-I_{p w}^2 / I_w
\end{aligned}
$$
with
$$
\begin{gathered}
I_x=\int_A y^2 \dA, 
\quad I_y=\int_A x^2 \dA, 
\quad I_w=\int_A \WarpTwistIesan^2 \dA \\
I_p=\int_A r^2 \dA, 
\quad I_{p p}=\int_A\left(x^2+y^2\right)^2 \dA \\
I_{p x}=\int_A y\,r^2 \dA, 
\quad I_{p y}=\int_A x \, r^2 \dA \\
I_{p w}=\int_A \WarpTwistIesan\, r^2 \dA \\
\beta_x=I_{p x} / I_x-2 y_0, 
\quad \beta_y=I_{p y} / I_y-2 x_0
\end{gathered}
$$

% The resultant torque $T$ comprises the linear uniform torque $G\Jsv \vartheta'$, the linear warping torque $-E\Jw \vartheta'''$, and the nonlinear torque $\tfrac12 E I_n \vartheta'''$ arising from the Wagner strain term \eqref{eq:wagner-gl-strain-pi}.

Under a constant torque $\TwistRxn$ the twist $\vartheta'$ is constant along the member and  ($\dagger$) admits the solution:
\begin{equation}
T(x)= \TwistRigidity\frac{\vartheta_L}{L}+\tfrac{1}{2} E I_n\left(\frac{\vartheta_L}{L}\right)^3
\label{eq:0015-solution}
\end{equation}
The section dimensions are $200 \times 10$\unit{\milli\meter} with constants 
%$J=66.667 \times 10^3 \mathrm{~mm}^4$, 
$\Jw\approx 0 \mathrm{~mm}^6$, $I_n=17.778 \times 10^9 \mathrm{~mm}^6$, and length $L=1,000 \mathrm{~mm}$.
