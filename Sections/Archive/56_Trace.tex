The \emph{Ieşan parameter vector} is defined as
\begin{equation}
\IesanVector \triangleq \begin{pmatrix} 
a_\varepsilon \\ \bm{a}_\Section \\ a_\vartheta + c_\vartheta \\ \bm{b}_\Section 
\end{pmatrix} \in \mathbb{R}^{6}
% \label{eq:def-iesan-params}
\end{equation}

%

augmented by an arbitrary rigid body motion \eqref{eq:general-solution}:
\begin{equation*}
\hat{\bm{u}}(\reflenx, \bm{r}) = \hat{\bm{u}}_{\rm P}(\reflenx, \bm{r}; \IesanVector) 
+ \hat{\bm{u}}_{\rm R}(\reflenx, \bm{r}; \bm{\delta})
\end{equation*}
where $\bm{\delta} = (\bm{\delta}_u, \bm{\delta}_\theta) \in \mathbb{R}^6$ 
represents the six rigid body modes. The stress field, and hence all resultants, 
depend only on $\IesanVector$.

The resultants are related to the Ieşan parameters through a linear map:
\begin{equation}
\begin{pmatrix} N \\ T \\ \ShearRxn \\ \FrameBasis \times \bm{M} \end{pmatrix}
= \IesanMatrix \, \IesanVector
% \label{eq:resultant-map}
\end{equation}
where the \emph{resultant matrix} $\IesanMatrix \in \mathbb{R}^{6 \times 6}$ 
encodes the Saint-Venant constitutive relations:
\begin{equation}
\IesanMatrix = -\begin{pmatrix}
EA & EA\CentroidLocation^\top & 0 & \mathbf{0}^\top \\
0 & \mathbf{0}^\top & G\Jo & G\,\FrameBasis \cdot \int_\Section \bm{r} \times 
\big(\nabla\bm{\WarpShearIesan} + \bm{\Pi}_{(b)}\big) \mathrm{d}A \\
\mathbf{0} & E\IntRGoRG & \mathbf{0} & \mathbf{0} \\
E\IntRGoRG \CentroidLocation & E\IntRGoRG & \mathbf{0} & \mathbf{0}
\end{pmatrix}
% \label{eq:def-R-matrix}
\end{equation}
The matrix $\IesanMatrix$ is invertible for any section with positive-definite 
inertia tensor $\IntRGoRG$.
\end{Quote}

\begin{Quote}
\begin{definition}
A \emph{trace operator} is a linear map 
$\mathcal{T}: H^1(\Section; \mathbb{R}^3) \to \mathbb{R}^6$ 
that assigns a translation-rotation pair from a displacement field over the section:
\begin{equation}
\mathcal{T}[\hat{\bm{u}}(\reflenx, \cdot)] = 
\begin{pmatrix} \bm{u}(\reflenx) \\ \bm{\theta}(\reflenx) \end{pmatrix}
\label{eq:def-trace}
\end{equation}
\end{definition}
\end{Quote}

\begin{definition}[Trace]
% \label{def:sv-trace}
Let $\mathcal{S}_{\mathrm{SV}}$ denote the six-dimensional vector space of 
Saint-Venant displacement fields for a given prismatic bar (modulo global rigid 
motions). 
A \emph{trace} for the Saint-Venant class is a linear map
\[
\mathcal{T} : \mathcal{S}_{\mathrm{SV}} \to C^1(I;\mathbb{R}^6),
\qquad
\hat{\bm{u}} \mapsto 
\begin{pmatrix}
\bm{u} \\[2pt] \bm{\theta}
\end{pmatrix},
\]
where $I \subset \mathbb{R}$ is the axial interval, and the image is interpreted 
as a pair of centerline fields
\[
\bm{u} : I \to \mathbb{R}^3,
\qquad
\bm{\theta} : I \to \mathbb{R}^3.
\]
We say that $\mathcal{T}$ is \emph{Saint-Venant linear} if it is linear on 
$\mathcal{S}_{\mathrm{SV}}$ in the sense that, for any 
$\hat{\bm{u}}_1,\hat{\bm{u}}_2 \in \mathcal{S}_{\mathrm{SV}}$ and 
$\alpha,\beta \in \mathbb{R}$,
\[
\mathcal{T}[\alpha \hat{\bm{u}}_1 + \beta \hat{\bm{u}}_2]
=
\alpha\,\mathcal{T}[\hat{\bm{u}}_1] + \beta\,\mathcal{T}[\hat{\bm{u}}_2].
\]
\end{definition}


\begin{definition}[Trace matrices]
\label{def:trace-matrices}
Fix a Saint-Venant linear trace $\mathcal{T}$ on $\mathcal{S}_{\mathrm{SV}}$, 
and suppose its associated beam strains satisfy the affine-in-$x$ structure
\begin{equation}
\bm{e}(x;\IesanVector)
=
\bm{e}_0(\IesanVector) + x\,\bm{e}_1(\IesanVector),
\qquad
\bm{e}_0,\bm{e}_1 : \mathbb{R}^6 \to \mathbb{R}^6 \ \text{linear}.
\label{eq:trace-affine-e}
\end{equation}

The \emph{trace matrices} of $\mathcal{T}$ are the unique matrices
$\TraceDeDpRoot,\TraceDeDpOne \in \mathbb{R}^{6\times 6}$ such that
\begin{equation}
\label{eq:def-T0-T1}
\bm{e}_0(\IesanVector) = \TraceDeDpRoot\,\IesanVector,
\qquad
\bm{e}_1(\IesanVector) = \TraceDeDpOne\,\IesanVector
\quad
\text{for all }\IesanVector\in\mathbb{R}^6.
\end{equation}

Equivalently, the stacked trace matrix
\begin{equation}
\label{eq:def-Tstack}
\mathbf{T}
\;\triangleq\;
\begin{pmatrix}
\TraceDeDpRoot \\[2pt]
\TraceDeDpOne
\end{pmatrix}
\in \mathbb{R}^{12\times 6}
\end{equation}
represents the linear map from Ieşan parameters to the pair of strain 
coefficients $(\bm{e}_0,\bm{e}_1)$:
\begin{equation}
\label{eq:Tstack-map}
\begin{pmatrix}
\bm{e}_0(\IesanVector) \\[2pt]
\bm{e}_1(\IesanVector)
\end{pmatrix}
=
\mathbf{T}\,\IesanVector.
\end{equation}
\end{definition}

\begin{remark}
The matrix $\TraceDeDpRoot$ carries the dependence of the \emph{constant} strain 
coefficients on the Ieşan parameters, while $\TraceDeDpOne$ encodes the 
\emph{linear} strain gradients in $x$ within the Saint-Venant class. Different 
traces (Cowper, section-average, energy-conjugate, gauge-shifted variants) are 
fully characterized at this level by different trace matrices 
$(\TraceDeDpRoot,\TraceDeDpOne)$.
\end{remark}

\begin{definition}[Warping index]
\label{def:warping-index}
Let $\mathcal{T}$ be a Saint-Venant linear trace whose beam strains satisfy 
the affine-in-$x$ structure \eqref{eq:trace-affine-e} for all 
$\IesanVector \in \mathbb{R}^6$, and let 
$\TraceDeDpRoot,\TraceDeDpOne,\mathbf{T}$ be the associated 
trace matrices from Definition~\ref{def:trace-matrices}.

The \emph{warping index} of the trace $\mathcal{T}$ is defined as
\begin{equation}
\label{eq:def-warp-index}
\WarpIndex
\;\triangleq\;
\dim \ker\!\big(\mathbf{T}\big)
\;=\;
6 - \operatorname{rank}\!\big(\mathbf{T}\big).
\end{equation}
Equivalently,
\[
\WarpIndex
=
\dim\left\{
\IesanVector \in \mathbb{R}^6 \;:\;
\bm{e}_0(\IesanVector) = \mathbf{0},\;
\bm{e}_1(\IesanVector) = \mathbf{0}
\right\}.
\]
\end{definition}

% \begin{remark}[Information content of the warping index]\label{rem:warping-index-interpretation}
The warping index $\WarpIndex$ measures how much information about the 
six-dimensional Saint-Venant parameter vector $\IesanVector$ is lost when passing to 
the one-dimensional beam strain field $\bm{e}(x)$.

\begin{itemize}
  \item If $\WarpIndex = 0$, then the stacked trace matrix has full column 
  rank and the map
  \[
  \IesanVector \;\mapsto\; (\bm{e}_0(\IesanVector),\bm{e}_1(\IesanVector))
  \]
  is injective. In this case, the full Saint-Venant state is uniquely 
  determined by the beam strain field $\bm{e}(x)$ (or equivalently, by its 
  coefficients $\bm{e}_0,\bm{e}_1$). A six-strain Cosserat beam model 
  \emph{can, in principle, represent the entire classical Saint-Venant class} 
  for that trace, and all local cross-sectional strains and warping amplitudes 
  can be recovered as linear functions of $(\bm{e}_0,\bm{e}_1)$.

  \item If $\WarpIndex > 0$, then there exist nonzero Ieşan parameter vectors 
  $\IesanVector$ such that $\bm{e}_0(\IesanVector) = \bm{e}_1(\IesanVector) = \mathbf{0}$, i.e. 
  the corresponding Saint-Venant solutions produce \emph{identically vanishing} 
  beam strains $\bm{e}(x) \equiv \mathbf{0}$. These directions in the 
  Saint-Venant parameter space are invisible to the six beam strain measures. 
  To represent the full classical Saint-Venant family with that trace, one 
  would require at least $\WarpIndex$ additional one-dimensional fields (beyond 
  $\bm{\gamma},\bm{\kappa}$) to parameterize the missing modes.
\end{itemize}
% \end{remark}

\begin{remark}
\label{rem:warping-index-implementation}
The warping index provides a quantitative diagnostic for a given trace:
\begin{enumerate}
  \item If $\WarpIndex = 0$, then a six-degree-of-freedom beam element using 
  only the Cosserat strains $\bm{e}(x)$ is, at least at the Saint-Venant 
  level, capable of uniquely encoding all classical torsion–extension–shear–
  bending states for that cross-section and trace. The section reconstruction 
  (from $\bm{e}(x)$ to local three-dimensional strains) can be implemented as 
  a purely local linear operation.

  \item If $\WarpIndex > 0$, then some Saint-Venant modes are not seen by 
  $\bm{e}(x)$ for that trace. A six-strain beam element cannot be fully 
  Saint-Venant exact in this setting; one must either accept a reduced model 
  that discards those modes, or enrich the beam kinematics with at least 
  $\WarpIndex$ additional generalized fields in $x$ to recover injectivity on 
  the Saint-Venant class.
\end{enumerate}
The detailed treatment of the cases $\WarpIndex=0$ and $\WarpIndex>0$ is 
deferred to later subsections.
\end{remark}
