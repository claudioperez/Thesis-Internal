\define@key{beamsec}{label/shape}[]{\def\beamsec@label@shape{#1}}

%% Kinematics

% Helper for property row
\newcommand{\beamsec@row}[3]{%
  \ifx#2\empty\else$#1$ & $\!\!=\!$ & \num{#2}\ifx#3\empty\else\;\si{#3}\fi \tabularnewline \fi
}

\newtcolorbox{beamsecbox}{
    colback=white,
    colframe=black!60,
    boxrule=0.4pt,
    arc=1pt,
    left=8pt,
    right=8pt,
    top=6pt,
    bottom=6pt,
}

\newcommand{\FrameSection}[1]{%
    % 1) Initialize all to empty
    \def\beamsec@name{}\def\beamsec@shape{}\def\beamsec@material{}%
    \def\beamsec@model{}
    % Labels
    \def\beamsec@label@section{}
    \def\beamsec@label@problem{}
    \def\beamsec@label@shape{}
    % Kinematics 
    \def\beamsec@kinematics@section{}
    \def\beamsec@kinematics@element{}
    % Section Geometry
    \def\beamsec@d{}
    \def\beamsec@b{}
    \def\beamsec@A{}
    \def\beamsec@Ay{}\def\beamsec@Az{}%
    \def\beamsec@Iy{}\def\beamsec@Iz{}%
    \def\beamsec@J{}\def\beamsec@ry{}\def\beamsec@rz{}%
    % Material
    \def\beamsec@E{}\def\beamsec@G{}\def\beamsec@nu{}
    \def\beamsec@Fy{}\def\beamsec@rho{}%
    \def\beamsec@unit@length{}\def\beamsec@unit@stress{}%

    % 2) Parse keys
    \setkeys{beamsec}{#1}%

    %
    \begin{beamsecbox}
        % Header
        \ifx\beamsec@label@problem\empty\else 
          {\textbf{Properties for Example~\ref{sec:\beamsec@label@problem}. (\texttt{\beamsec@label@problem})}}\par
        \fi
        % \ifx\beamsec@shape\empty
        %     \ifx\beamsec@material\empty\else
        %         \beamsec@material\par
        %     \fi
        % \else
        %     \beamsec@shape
        %     \ifx\beamsec@material\empty\else\,·\,\beamsec@material\fi
        %     \par
        % \fi
        \vspace{4pt}
        \hrule
        \vspace{6pt}
        %
        % side-by-side tables
        \begin{minipage}[t]{0.3\linewidth}
          \textit{Kinematics} \\[2pt]
          \begin{tabular}{@{}rl}
             Section: & \beamsec@kinematics@section \\
             Element: & \beamsec@kinematics@element \\
          \end{tabular}
        \end{minipage}
        \hfill
        \begin{minipage}[t]{0.3\linewidth}
            
            \ifx\beamsec@shape\empty
                \textit{Section}
            \else 
                \textit{\beamsec@shape}
            \fi
            \ifx\beamsec@label@shape\empty\else 
              (\beamsec@label@shape)
            \fi
            \\[2pt]
            \begin{tabular}{@{}lcl@{}}
                \beamsec@row{d}{\beamsec@d}{\beamsec@unit@length\squared}
                \beamsec@row{b}{\beamsec@b}{\beamsec@unit@length\squared}
                \beamsec@row{A}{\beamsec@A}{\beamsec@unit@length\squared}
                \beamsec@row{A_y}{\beamsec@Ay}{\beamsec@unit@length\tothe{2}}
                \beamsec@row{A_z}{\beamsec@Az}{\beamsec@unit@length\tothe{2}}
                \beamsec@row{I_y}{\beamsec@Iy}{\beamsec@unit@length\tothe{4}}
                \beamsec@row{I_z}{\beamsec@Iz}{\beamsec@unit@length\tothe{4}}
                \beamsec@row{J}{\beamsec@J}{{\beamsec@unit@length}\tothe{4}}
            \end{tabular}
        \end{minipage}%
        \hfill
        \begin{minipage}[t]{0.3\linewidth}
            \textit{Material}\\[2pt]
            \begin{tabular}{@{}rcl@{}}
                \beamsec@row{E}{\beamsec@E}{\beamsec@unit@stress}
                \beamsec@row{G}{\beamsec@G}{\beamsec@unit@stress}
                % \beamsec@row{\nu}{\beamsec@nu}{}
                \beamsec@row{F_y}{\beamsec@Fy}{\beamsec@unit@stress}
                % \beamsec@row{\rho}{\beamsec@rho}{}
            \end{tabular}
        \end{minipage}
    \end{beamsecbox}
}
