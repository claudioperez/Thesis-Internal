\begin{proposition}[Properties of the Orthogonal Decomposition]\label{cor:decomp-properties}
Any square-integrable field $\hat{\bm u} : \Section \to \mathbb{R}^3$ admits the unique decomposition
\begin{equation*}
    \hat{\bm u} = \underbrace{\SectionAverage{\bm u}}_{\bm u^{\mathrm{tr}}} 
          + \underbrace{\RotationAverage{\bm u} \times \CoordinateCentroid}_{\bm u^{\mathrm{rot}}} 
          + \underbrace{\bm{w}}_{\mathrm{warping}},
    \qquad
    \CoordinateCentroid := \bm{r} - \CentroidLocation,
\end{equation*}
where the warping field satisfies $\SectionAverage{\bm{w}} = \mathbf{0}$ and $\RotationAverage{\bm{w}} = \mathbf{0}$. This decomposition enjoys the following properties:

\begin{enumerate}[label=(\roman*)]
    \item \textbf{Uniqueness.} The decomposition is unique: if 
    $\bm u = \bm a_1 + \bm\omega_1 \times \CoordinateCentroid + \bm u_1' = \bm a_2 + \bm\omega_2 \times \CoordinateCentroid + \bm u_2'$
    with $\SectionAverage{\bm u_i'} = \mathbf{0}$ and $\RotationAverage{\bm u_i'} = \mathbf{0}$, then $\bm a_1 = \bm a_2$, $\bm\omega_1 = \bm\omega_2$, and $\bm u_1' = \bm u_2'$.

    \item \textbf{Orthogonality.} The three components are mutually $L^2(\Section)$-orthogonal:
    \begin{equation*}
        \int_{\Section} \bm u^{\mathrm{tr}} \cdot \bm u^{\mathrm{rot}} \dA = 0,
        \qquad
        \int_{\Section} \bm u^{\mathrm{tr}} \cdot \bm{w} \dA = 0,
        \qquad
        \int_{\Section} \bm u^{\mathrm{rot}} \cdot \bm{w} \dA = 0.
    \end{equation*}

    \item \textbf{Pythagorean identity.} The $L^2$-norm decomposes as
    \begin{equation*}
        \int_{\Section} |\bm u|^2 \dA 
        = A \bigl\|\SectionAverage{\bm u}\bigr\|^2 
        + \RotationAverage{\bm u} \cdot \CentroidStaticInertia \, \RotationAverage{\bm u} 
        + \int_{\Section} |\bm{w}|^2 \dA.
    \end{equation*}

    \item \textbf{Best approximation.} The rigid field $\bm u^{\mathrm{rig}} := \SectionAverage{\bm u} + \RotationAverage{\bm u} \times \CoordinateCentroid$ is the unique minimizer of
    \begin{equation*}
        \min_{\bm a,\, \bm\omega \in \mathbb{R}^3} \int_{\Section} \bigl\|\bm u - \bm a - \bm\omega \times \bm{r}\bigr\|^2 \dA.
    \end{equation*}

    \item \textbf{Characterization of the warping space.} The warping space $\mathcal{W} := \ker(\SectionAverage{\cdot}) \cap \ker(\RotationAverage{\cdot})$ is the $L^2$-orthogonal complement of the six-dimensional space of rigid fields $\mathcal{R} := \{\bm a + \bm\omega \times \CoordinateCentroid : \bm a, \bm\omega \in \mathbb{R}^3\}$.

    \item \textbf{Projection operators.} The linear operators
    \begin{equation*}
        \mathcal{P}^{\mathrm{tr}}[\bm u] := \SectionAverage{\bm u},
        \qquad
        \mathcal{P}^{\mathrm{rot}}[\bm u] := \RotationAverage{\bm u} \times \CoordinateCentroid,
        \qquad
        \mathcal{P}^{\mathrm{warp}} := \mathcal{I} - \mathcal{P}^{\mathrm{tr}} - \mathcal{P}^{\mathrm{rot}}
    \end{equation*}
    are $L^2$-orthogonal projections satisfying $\mathcal{P}^{\mathrm{tr}} + \mathcal{P}^{\mathrm{rot}} + \mathcal{P}^{\mathrm{warp}} = \mathcal{I}$, $(\mathcal{P}^{\alpha})^2 = \mathcal{P}^{\alpha}$, and $\mathcal{P}^{\alpha}\mathcal{P}^{\beta} = 0$ for $\alpha \neq \beta$.
\end{enumerate}
\end{proposition}

\begin{proof}
We establish each property in turn.

\textit{(i) Uniqueness.} 
Applying $\SectionAverage{\cdot}$ to both decompositions and using $\SectionAverage{\CoordinateCentroid} = \mathbf{0}$ yields $\bm a_1 = \bm a_2$. Applying $\RotationAverage{\cdot}$ and invoking the projection property $\RotationAverage{\bm\omega \times \bm{r}} = \bm\omega$ yields $\bm\omega_1 = \bm\omega_2$. The equality $\bm u_1' = \bm u_2'$ follows by subtraction.

\textit{(ii) Orthogonality.} 
For the first identity, $\int_{\Section} \bm u^{\mathrm{tr}} \cdot (\bar{\bm\theta} \times \CoordinateCentroid) \dA = \bm u^{\mathrm{tr}} \cdot (\bar{\bm\theta} \times \int_{\Section} \CoordinateCentroid \dA) = \mathbf{0}$. 
For the second, $\int_{\Section} \bm u^{\mathrm{tr}} \cdot \bm{w} \dA = A\, \bm u^{\mathrm{tr}} \cdot \SectionAverage{\bm{w}} = \mathbf{0}$. 
For the third, use the scalar triple product identity:
\begin{equation*}
    \int_{\Section} (\bar{\bm\theta} \times \CoordinateCentroid) \cdot \bm{w} \dA 
    = \bar{\bm\theta} \cdot \int_{\Section} \CoordinateCentroid \times \bm{w} \dA 
    = \bar{\bm\theta} \cdot \CentroidStaticInertia \, \RotationAverage{\bm{w}} = \mathbf{0}.
\end{equation*}

\textit{(iii) Pythagorean identity.} 
Expand $|\bm u|^2 = |\bm u^{\mathrm{tr}} + \bm u^{\mathrm{rot}} + \bm{w}|^2$; by (ii), all cross terms vanish upon integration. For the rotational term, use $|\bm a \times \bm b|^2 = \bm a \cdot [(|\bm b|^2\mathbf{1} - \bm b \otimes \bm b)\cdot\bm a]$ to obtain
\begin{equation*}
    \int_{\Section} |\bar{\bm\theta} \times \CoordinateCentroid|^2 \dA 
    = \bar{\bm\theta} \cdot \biggl(\int_{\Section} \bigl(|\CoordinateCentroid|^2 \mathbf{1} - \CoordinateCentroid \otimes \CoordinateCentroid\bigr) \dA\biggr) \cdot \bar{\bm\theta}
    = \bar{\bm\theta} \cdot \CentroidStaticInertia \, \bar{\bm\theta}.
\end{equation*}

\textit{(iv) Best approximation.} 
For any rigid field $\bm v^{\mathrm{rig}} = \bm a + \bm\omega \times \CoordinateCentroid$, write
$\|\bm u - \bm v^{\mathrm{rig}}\|^2 = \|(\bm u^{\mathrm{rig}} - \bm v^{\mathrm{rig}}) + \bm{w}\|^2$.
Since $\bm u^{\mathrm{rig}} - \bm v^{\mathrm{rig}} \in \mathcal{R}$ and $\bm{w} \in \mathcal{R}^\perp$, property (ii) gives
$\|\bm u - \bm v^{\mathrm{rig}}\|^2 = \|\bm u^{\mathrm{rig}} - \bm v^{\mathrm{rig}}\|^2 + \|\bm{w}\|^2 \geq \|\bm{w}\|^2$,
with equality if and only if $\bm v^{\mathrm{rig}} = \bm u^{\mathrm{rig}}$.

\textit{(v) Characterization of the warping space.} 
Let $\bm{w} \in \mathcal{W}$. For any $\bm v = \bm a + \bm\omega \times \CoordinateCentroid \in \mathcal{R}$,
\begin{equation*}
    \int_{\Section} \bm{w} \cdot \bm v \dA 
    = A\,\SectionAverage{\bm{w}} \cdot \bm a + \bm\omega \cdot \CentroidStaticInertia \, \RotationAverage{\bm{w}} = 0,
\end{equation*}
so $\mathcal{W} \subseteq \mathcal{R}^\perp$. Conversely, if $\bm{w} \perp \mathcal{R}$, testing against constant fields gives $\SectionAverage{\bm{w}} = \mathbf{0}$, and testing against pure rotations gives $\RotationAverage{\bm{w}} = \mathbf{0}$ (since $\CentroidStaticInertia$ is positive definite), so $\mathcal{R}^\perp \subseteq \mathcal{W}$.

\textit{(vi) Projection operators.} 
The identity $\mathcal{P}^{\mathrm{tr}} + \mathcal{P}^{\mathrm{rot}} + \mathcal{P}^{\mathrm{warp}} = \mathcal{I}$ holds by definition. Idempotence of $\mathcal{P}^{\mathrm{tr}}$ follows from $\SectionAverage{\SectionAverage{\bm u}} = \SectionAverage{\bm u}$. For $\mathcal{P}^{\mathrm{rot}}$, the projection property $\RotationAverage{\bm\omega \times \bm{r}} = \bm\omega$ gives $(\mathcal{P}^{\mathrm{rot}})^2 = \mathcal{P}^{\mathrm{rot}}$. 
Mutual annihilation follows from $\SectionAverage{\CoordinateCentroid} = \mathbf{0}$ and annihilation of constants by $\RotationAverage{\cdot}$. Idempotence of $\mathcal{P}^{\mathrm{warp}}$ then follows algebraically.
\end{proof}
