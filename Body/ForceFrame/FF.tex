
\section{Newton linearization and Schur complement}

Each Newton step seeks \((\dd \bm{q},\dd \bm{e})\) such that

\[
\residv[\bm{e}+\dd \bm{e}]\approx 0,\qquad \resids(x_{\ell};\bm{q}+\dd \bm{q},\,\bm{e}+\dd \bm{e})\approx 0.
\]

Linearizing \eqref{eq:def-mf-gv} furnishes:
\[
\residv[\bm{e}+\dd \bm{e}] = \residv[\bm{e}]-\int_0^L \ForceInterp^{\top}\,\dd \bm{e}\,\dd\reflenx.
\]

Setting \(\residv[\bm{e}+\dd \bm{e}]=\mathbf{0}\) gives

\begin{equation}
\int_0^L \ForceInterp^{\top}\,\dd \bm{e}\,\dd\reflenx = \residv[\bm{e}].
\label{eq:C}
\end{equation}

Linearizing \eqref{eq:def-mf-gs} furnishes:
\[
\resids(x;q+\dd \bm{q},e+\dd \bm{e})\approx \resids(x;q,e) + \ForceInterp(x)\dd \bm{q} - \Ks(x)\dd \bm{e}(x)
\]

where

\begin{equation}
\Ks(x):=\frac{\partial s}{\partial e}(x),\qquad \mathbf{F}_{s}(x):=\Ks^{-1}(x).
\label{eq:def-section-flexibility}
\end{equation}

Setting the linearization to zero:

\begin{equation}
\ForceInterp(x)\dd \bm{q} - \Ks(x)\dd \bm{e}(x) = -\resids(x;q,e).
\end{equation}

Solve for \(\dd \bm{e}(x)\):

\begin{equation}
\dd \bm{e}(x) = \mathbf{F}_{s}(x)\,\ForceInterp(x)\,\dd \bm{q} \;+\; \mathbf{F}_{s}(x)\,\resids(x;q,e).
\label{eq:S}
\end{equation}

Now substitute \eqref{eq:S} into \eqref{eq:C}:

\[
\int_0^L \ForceInterp^{\top} \mathbf{F}_{s} \ForceInterp\,\dd\reflenx\;\dd \bm{q} 
\;+\;\int_0^L \ForceInterp^{\top} \mathbf{F}_{s} \resids\,\dd\reflenx \;=\; \residv[\bm{e}].
\]

Define

\begin{equation}
\mathbf{F}_q := \int_0^L \ForceInterp^{\top} \mathbf{F}_{s} \ForceInterp\,\dd\reflenx,
\qquad
\tilde{\bm{g}}_{\rm v} := \residv - \int_0^L \ForceInterp^{\top} \mathbf{F}_{s} \resids\,\dd\reflenx
% \quad
% v_r := \int_0^L \ForceInterp^{\top} \mathbf{F}_{s} \resids\,\dd\reflenx
% = \int_0^L \ForceInterp^{\top} \mathbf{F}_{s}\big(\inters-s(e)\big)\,\dd\reflenx.
\label{eq:def-mv-flex}
\end{equation}

Then the staggered Newton step is

\begin{equation}
\mathbf{F}_q\,\dd \bm{q} = \tilde{\bm{g}}_{\rm v} %= \residv[\bm{e}]-v_r.
\label{eq:mf-dq}
\end{equation}

\subsubsection{Line Search}
The basic forces \(\bm{q}\) are updated by scaling the Newton step furnished by \eqref{eq:mf-dq}:
\begin{equation}
\bm{q} \leftarrow \bm{q} + \eta\, \dd \bm{q}
\label{eq:def-line-search}
\end{equation}
where \(\eta\) is determined by a line search algorithm. 
A backtracking line search is performed with 
\begin{equation}
\phi(\eta) = \|\residv\|
\label{eq:def-line-merit}
\end{equation}
