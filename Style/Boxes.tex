\newtcolorbox{marsdenbox}[1][]{% 
  enhanced jigsaw,
  sharp corners,
  colbacklower=white,%black!75
  title={#1},
  colback=white,
  borderline={1pt}{-2pt}{black},
  fontupper={\setlength{\parindent}{20pt}}%, #2
}

\newtcolorbox[%
% use counter=myambox, %
auto counter,%
list inside=ambox,%
number within=section,%
%crefname=Box,
number freestyle={\noexpand\thesection.\noexpand\arabic{\tcbcounter}}
]%
{ambox}[2][]%
{%
  enhanced jigsaw,
  sharp corners,
  label=#2,
  colbacklower=white,%black!75
  title={Box~\thetcbcounter: #1},
  list text=#1,
  boxed title style={%
    enhanced,% 
    size=small,boxrule=0pt,arc=0pt,%
    colback=white,%
    colframe=white,%
    toprule=1pt,%
    leftrule=1pt,rightrule=1pt%
  },%
  colback=white,
  borderline={1pt}{-2pt}{black},
  before={\noindent}
  % fontupper={\setlength{\parindent}{20pt}}%, #2
}
