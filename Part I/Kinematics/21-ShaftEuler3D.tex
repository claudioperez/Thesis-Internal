
This model is discussed by \cite{ziegler1977principles}, \cite{bažant2010stability} and many more. It should exhibit hockling.
Recall the approximate rotation field given by \cref{eq:stiff-euler-theta}:
$$
\bm{\theta} \approx \vartheta \FrameBasis  
+ \FrameBasis \times\bar{\bm{u}}^{\prime} 
+ \frac{1}{2} (\FrameBasis \times\bar{\bm{u}}^{\prime}) \times \vartheta \FrameBasis 
$$
This produces strains given by \cref{eq:stiff-euler-strain}.
Dropping terms furnishes:
\begin{equation}
\begin{aligned}
\ShearStrain &= \left(\FrameBasis \otimes\FrameBasis  + \frac{1}{2}\FrameBasis \otimes\bar{\bm{u}}^{\prime}\right)\bm{u}^{\prime} \\
\CurveStrain &= \vartheta^{\prime} \FrameBasis  
+ \FrameBasis \times\bar{\bm{u}}^{\prime\prime} 
+ \frac{1}{2}(\FrameBasis \times\bar{\bm{u}}^{\prime})\times(\FrameBasis \times\bar{\bm{u}}^{\prime\prime}) 
%+ (\FrameBasis \times\bar{\bm{u}}^{\prime\prime})\times(\vartheta \FrameBasis )
\end{aligned}
\end{equation}

\begin{equation}
\bm{e} = \begin{pmatrix}
\epsilon \\ 
\tau \\
\kappa_y \\
\kappa_z \\ 
\end{pmatrix}
=\left(\begin{array}{c}
u^{\prime}+\frac{1}{2}\left(\left(u_y^{\prime}\right)^2+\left(u_z^{\prime}\right)^2\right) \\ % +\left(z_s u_y^{\prime}-y_s u_z^{\prime}\right) \vartheta^{\prime} \\
\vartheta^{\prime} +\frac{1}{2}\left(u_{z}^{\prime} u_{y}^{\prime\prime}-u_{y}^{\prime} u_{z}^{\prime\prime}\right)\\
-u_z^{\prime \prime} \\ % +u_y^{\prime \prime} \vartheta \\
u_y^{\prime \prime} \\ % +u_z^{\prime \prime} \vartheta \\
\end{array}\right)
\end{equation}

\subsubsection{Equilibrium}

\[
\bm{\sigma} = \bm{A}^{\mathrm{t}} \bm{s} = \begin{pmatrix}
N (\FrameBasis  + \bar{\bm{u}}') + \frac{1}{2} T \FrameBasis  \times \bar{\bm{u}}'' \\
T \\ 
\bar{\CurveResult} + \frac{1}{2} T \bar{\bm{u}}' \\
 0
\end{pmatrix}
\]

\begin{equation*}
\begin{aligned}
&= 
-\left[\begin{array}{cccc}
\bm{1}\partial & 0 & \FrameBasis^{\times}\partial^2 & 0\\
0 & \partial & 0 & 1
\end{array}\right]
\begin{pmatrix}
N (\FrameBasis  + \bar{\bm{u}}') + \frac{1}{2} T \FrameBasis  \times \bar{\bm{u}}'' \\
T \\

\end{pmatrix} \\

T' \\ 
\end{pmatrix} \\
\end{aligned}
\end{equation*}
