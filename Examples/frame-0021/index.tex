\subparagraph{Context}
This example investigates non-uniform torsional warping. 
The analytic solution \eqref{eq:0028-gen-rot} to the beam governing equations (Box~\ref{box:exact-shear-svq}) is used as reference. 


A similar example was considered by \cite{lecorvec2012nonlinear}, where 16 warping degrees of freedom were required. 
Here only one warping degree of freedom is used. Excellent agreement with the analytic solution is obtained using both the \ForceFrame{} and \ExactFrame{} elements.

\subsubsection*{General Solution}

Consider a cross section satisfying \(\CenterTwistLocation \equiv \CentroidLocation\). 
Align this point with the origin of the plane coordinate system, \(\PlanePosition = \bm{0}\). 

% The beam ends can have either an imposed rotation $\vartheta=\vartheta_0$ or an imposed torque $T=\TwistRxn$. 
% At the same time they have either an imposed warping $\vartheta'=\vartheta_0^{\prime}$ or an imposed bimoment $B=B_0$. 
\begin{enumerate}
\item $\mathrm{D}_{\vartheta}$ twisting angle restrained $\vartheta=0$
\item $\mathrm{D}_{\alpha}$ warping restrained $\alpha=0$ or $\vartheta^{\prime}=0$
\item Warping free $B \,(= \alpha') =0$ or $\theta^{\prime \prime}=0$
\item Twisting free $\vartheta^{\prime \prime \prime}-\lambda^2 \vartheta^{\prime}=0$ %$\qquad\implies T=B'$
\end{enumerate}

\paragraph{Kinematics.}
The rotation \(\bm{\Lambda}\)
can be represented with only a single parameter, say \(\vartheta\), so that:
\begin{equation}
\begin{aligned}
\bm{\Lambda} &= \operatorname{Exp} \vartheta \FrameBasis 
\end{aligned}
\qquad\text{and}\qquad 
\begin{aligned}
\bm{\kappa}^{\times} &= \bm{\Lambda}^{\prime}\bm{\Lambda}^{\mathrm{t}} 
= \vartheta^{\prime} \, \FrameBasis^{\times}
\end{aligned}
\label{eq:rotation-field}
\end{equation}
from which we identify \(\Twist = \vartheta'\). 
Under this motion the deformed (spatial) axial director \(\frameBasis\) remains equivalent to the reference axial director \(\FrameBasis\) (that is, \(\frameBasis = \FrameRotation \FrameBasis \equiv \FrameBasis\)). 
The compatible linearized strain tensor \(\hat{\StrainTensor}\) is characterized by the strain vector \(\hat{\bm{\varepsilon}}\) of \eqref{eq:wagner-gl-strain-pi} which takes the form:
%
\begin{equation}
\begin{aligned}
\hat{\bm{\varepsilon}} 
&= \alpha' \WarpSV\,\FrameBasis + \Twist \FrameBasis \times (\PlanePosition-\tilde{\bm{\SectionLocation}}) + \alpha \nabla  \WarpSV \\
% &= \alpha' \WarpSV\,\FrameBasis +  \Twist \, \bigl(\nabla \WarpSV + \FrameBasis\times(\PlanePosition - \tilde{\bm{\SectionCoordinate}})\bigr) - (\Twist - \alpha) \nabla \WarpSV
\end{aligned}
\label{eq:torsion-strain}
\end{equation}
% the infinitesimal strain vector
% \begin{equation*}
% \bm{\varepsilon} = \vartheta' \FrameBasis \times (\SectionCoordinate - \bm{\SectionLocation}) + \alpha' \varphi \FrameBasis + \alpha \nabla \varphi

\paragraph{Constitution.}
For \(\bm{\gamma} = \bm{0}\) we have \(\bm{n} = N \, \FrameBasis\). 
The moment has only a torsional component
$$
\begin{aligned}
\bm{m}\cdot \FrameBasis 
&= \, \bar{G}I \Twist - \bar{G}\Jv \alpha - \bar{G}(\Jv - \EnhanConstFA \, \Jw) \EnhanParam
\end{aligned}
$$
%

\paragraph{Equilibrium.}
The moment equilibrium equation \eqref{eq:finite-equilibrium}$_2$ is stated in terms of the spatial moment \((\FrameRotation \bm{m})'\). 
However, for the rotation field \eqref{eq:rotation-field}, \(\FrameRotation\FrameBasis = \FrameBasis\) and the resulting equation exactly matches the linear theory:
\begin{equation*}
\begin{array}{rlcl}
\FrameBasis \cdot (\FrameRotation \bm{m})' &= \bigl(\bar{G} \Io \Twist - \bar{G}\Jv \alpha - (\Jv - \EnhanConstFA \, \Jw) \EnhanParam \bigr)' &=& 0 \\
\bm{w}' - \bm{v} &= \bar{E}\Jw \alpha'' - \bar{G}(\Jv (\alpha - \Twist) + \tilde{J} \EnhanParam) &=& 0
\end{array}
\end{equation*}
Note that \(\bm{x}' = \FrameBasis\) so that \(\FrameBasis\cdot \bm{x}' \times (\FrameRotation \bm{n})= \bm{0}\).

where:
\begin{equation}
\begin{aligned}
\Io &\triangleq \FrameBasis \cdot \LinearMM \FrameBasis \\
\bar{\eta} &\triangleq \frac{\Jv - \EnhanConstFA \Jw}{\Jv - \EnhanConstFA (2 \Jw - \EnhanConstFA \Js)} \\
\tilde{J} &\triangleq (\WarpSV,\, \WarpSV + \EnhanConstFA \WarpFA)_G = \Jv - \EnhanConstFA \Jw \\
\eta &= \frac{\Io - \tilde{J}\tilde{\eta}}{\Jv - \tilde{J}} = \frac{\Io - \tilde{J}\tilde{\eta}}{\beta \Jw} \\
J_{\eta} &= \Jv - \bar{\eta}\tilde{J}
\end{aligned}
\label{eq:enhan-param}
\end{equation}
%
The strain vector is:
%
\begin{equation}
\begin{aligned}
\bm{\varepsilon} 
% &= \hat{\StrainVector} + \tilde{\StrainVector} \\
% &= \underbrace{\alpha' \WarpSV\, \FrameBasis + \Twist\, \FrameBasis \times \PlanePosition + \alpha \nabla \WarpSV}_{\hat{\StrainVector}} + \EnhanShape \EnhanParam  \\
&= \alpha' \WarpSV\, \FrameBasis + \Twist\, \FrameBasis \times \PlanePosition + \alpha \nabla \WarpSV + (\Twist - \alpha) \bar{\eta} (\nabla \WarpSV + \EnhanConstFA \nabla \WarpFA) \\
% &= \alpha' \WarpSV\, \FrameBasis + \Twist\, \FrameBasis \times \PlanePosition  - \alpha  \EnhanConstFA \nabla \WarpFA + \Twist (\nabla \WarpSV + \EnhanConstFA \nabla \WarpFA) \\
% &= \alpha' \WarpSV\, \FrameBasis + \Twist \, (\FrameBasis \times \PlanePosition  + \nabla \WarpSV) + (\Twist - \alpha)\EnhanConstFA \nabla \WarpFA \\
\end{aligned}
\label{eq:26}
\end{equation}
%
\paragraph{Solution.}
The moment equilibrium equation allows one to write:
\begin{equation*}
    \alpha (\reflenx)= \eta \bigl(
     \vartheta^{\prime}(\reflenx) - C\bigr)
\end{equation*}
This allows the bimoment and bishear to be expressed in terms of the twist rate \(\Twist\):
\begin{equation*}
B = \bar{E}\Jw \eta \Twist' 
\qquad\text{and}\qquad
Q = \bar{G}(\Jv + \tilde{J}\bar{\eta})(1 - \eta) \Twist
\end{equation*}
and from equilibrium of the bimoment
\begin{equation}
\bar{E}\Jw \eta \Twist'' - \bar{G}(\Jv + \tilde{J}\bar{\eta})(1 - \eta) \Twist = 0
\end{equation}
%
In standard form:
\begin{equation*}
\vartheta''' - \lambda^2 \vartheta' = C,
\qquad\text{with}\quad
\lambda \triangleq \sqrt{\frac{\bar{G}J_{\eta}}{\bar{E}\Jw}\frac{1 - \eta}{\eta}}
\end{equation*}
%
This furnishes the general solution
\begin{equation}
\begin{aligned}
\vartheta(\reflenx)= \frac{\TwistRxn}{\lambda \eta \, \TwistRigidity}\left(
 C_1 + \lambda\eta \, \reflenx +  C_2 \cosh \lambda \reflenx + C_3 \sinh \lambda \reflenx
\right)
\end{aligned}
\label{eq:0028-gen-rot}
\end{equation}

\subparagraph{Case C: Fixed-Fixed}
Warping is prevented at both sides of a beam furnishing the boundary conditions
$$
\begin{aligned}
& \left.\vartheta\right|_{x=0}=\vartheta_i,\quad \left.\vartheta' \right|_{x=0}=0 \\
& \left.\vartheta\right|_{x=L}=\vartheta_j,\quad \left.\vartheta' \right|_{x=L}=0
\end{aligned}
$$
%
The solution for a constant distributed torque \(m_x\) is
\begin{equation*}
\begin{aligned}
\vartheta&=\left(\vartheta_j-\vartheta_i\right) \frac{\lambda x+1+\left(\lambda x-1\right) e^\beta-e^{\lambda x}+e^{\beta\left(1-\frac{x}{L}\right)}}{\beta+2+(\beta-2) e^\beta} \\
&+m_x G J \frac{\lambda x+1-\beta \frac{x^2}{L^2}-\left(\lambda x-1-\beta \frac{x^2}{L^2}\right) e^\beta-e^{\lambda x}-e^{\beta\left(1-\frac{x}{L}\right)}}{2 \beta L^2\left(1-e^\beta\right)}
\end{aligned}
\end{equation*}
Using, $T_i=-\left.T\right|_{x=0}, B_i=\left.B\right|_{x=0}, T_j=\left.T\right|_{x=L}$ and $B_2=-\left.B\right|_{x=L}$ the element stiffness matrix is

$$
\begin{pmatrix}
T_i \\ T_j
\end{pmatrix}
=\frac{\beta\left(1+e^\beta\right)}{\beta+2+(\beta-2) e^\beta}
\frac{G J}{L}
\begin{bmatrix}
 1 & -1 \\
-1 &  1
\end{bmatrix}
\begin{pmatrix}
\vartheta_i \\ \vartheta_j
\end{pmatrix}
+ m_x \begin{pmatrix}
-\frac{1}{2} L \\
-\frac{1}{2} L
\end{pmatrix}
$$

$$
\begin{aligned}
& B_i=-T \frac{L}{\beta} \tanh \left(\frac{1}{2} \beta\right)+m_x \frac{L^2}{\beta^2} \frac{2 \beta e^\beta+1-e^{2 \beta}}{1-e^{2 \beta}} \\

\end{aligned}
