A general procedure for implementing the Class~I traces parameterized by \cref{tab:trace-consts} in the setting of \cref{sec:genwarp} will be developed in \cref{sec:trace-shapes}. 
However, first two special cases of the Class~I traces are investigated.

\subsection{Class II: Equilibrium Consistency}\label{sec:class-ii}
The traces that fall under Class I are capable of exactly representing Saint~Venant's solutions in a standard finite element context, but without further restriction they may still exhibit certain undesirable properties.
% In particular, [TODO: elaborate]
% \begin{itemize}
%     \item Tangent symmetry
%     \item Nodal loads
% \end{itemize}

The strain measures \(\TraceStrain\) associated to a trace \(\mathcal{T}\) define work conjugate resultants \(\TraceStress\) through the relation 
\[
\left<\TraceStress,\,\delta\TraceStrain\right> = \left<\StressVector,\,\delta\StrainVector\right>
\]
Thus the conjugate resultants \(\TraceStress\) are generally distinct from the classical resultants \(\snm\) of \eqref{eq:resultant-defs} and do not necessarily follow the evolution \eqref{eq:resultant-evolution}. 
Thus a section constitutive model that returns \(\snm\) in the setting of Box~\ref{box:basic-shear} may be variationally incompatible, but one returning \(\TraceStress\) may violate continuum consistency. 
The former is not ideal, but nonetheless will produce a continuum exact representation as demonstrated in Example~\ref{sec:frame-0012}. 
The latter would appear unjustifiable unless it can be shown that the conjugate resultants of the trace \(\TraceStress\) satisfy the evolution \eqref{eq:equilibrium-ode}. 
Thus define Class~II as the subset of Class~I traces for which this holds. 

\begin{definition}[Class II]\label{def:class-ii}
A Class I trace \(\mathcal{T}\) is a \textbf{Class II} trace if the energy-conjugate resultants $\TraceStress$ satisfy the same equilibrium as the classical resultants \eqref{eq:equilibrium-ode}:
\begin{equation}
\TraceStress' = \mathbb{J}\,\TraceStress.
\label{eq:sbar-equilibrium}
\end{equation}
\end{definition}

Evidently the conjugate resultants \(\TraceStress\) will depend linearly on the trace deformations \(\TraceStrain\) through a linear operator \(\bar{\mathbf{C}}_T(x)\) defined by:
% The conjugate stiffness is constructed from the material strain energy:
\begin{equation}
\mathcal{U}(x) \coloneq \tfrac{1}{2}\int_\Section \bm{\varepsilon} \cdot \mathbb{C}\, \bm{\varepsilon} \dA 
= \tfrac{1}{2}\TraceStrain \cdot \bar{\mathbf{C}}_T(x) \,\TraceStrain,
\label{eq:energy-conjugacy-def}
\end{equation}

\begin{remark}
The \emph{conjugate resultant stiffness} \(\bar{\mathbf{C}}_T(x)\) is called \emph{prismatic} if it is independent of \(\reflenx\). 
Evidently, through inheritance of the Class~I property \eqref{eq:req-c1}, every \(\bar{\mathbf{C}}_T(x)\) in Class~II is prismatic. 
%
Thus prismatic $\TraceStiffNM$ implies prismatic $\bar{\mathbf{C}}_T$ but the converse does not hold. 
% Prismatic $\bar{\mathbf{C}}_T$ with non-prismatic 
% $\TraceStiffNM$ requires $\TraceDpDeRoot\TraceDeDpOne - \IesanEvolveSP \neq \mathbf{0}$ 
% but $\IesanEnergy$-skew. This is an exceptional situation. 
\end{remark}

% \begin{proposition}[S2. Equilibrium-Conjugate]
% \label{prop:equilibrium-compatibility}
% Let $\mathbf{C}_T$ and $\bar{\mathbf{C}}_T$ be prismatic. 
The condition 
\eqref{eq:sbar-equilibrium} holds if and only if
\begin{equation}
[\bar{\mathbf{G}}, \mathbb{J}] = \mathbf{0}, 
\quad \text{with}\quad
\bar{\mathbf{G}} \coloneq \bar{\mathbf{C}}_T\,\mathbf{C}_T^{-1}
\qquad\forall x.
\label{eq:commutator-condition}
\end{equation}
where \([\cdot,\,\cdot]\) is the commutator \citep{abraham1988manifolds}. 
% \end{proposition} 
% \begin{proof}
A proof is as follows. 
With prismatic stiffnesses, $\TraceStress = \bar{\mathbf{C}}\,\bm{e}$ and 
$\snm = \mathbf{C}_T\,\bm{e}$, so $\TraceStress = \bar{\mathbf{G}}\,\snm$. 
From \eqref{eq:equilibrium-ode}:
\begin{equation*}
\TraceStress' = \bar{\mathbf{G}}\,\snm' = \bar{\mathbf{G}}\,\mathbb{J}\,\snm 
= \bar{\mathbf{G}}\,\mathbb{J}\,\bar{\mathbf{G}}^{-1}\,\TraceStress.
\end{equation*}
The equality \eqref{eq:sbar-equilibrium} thus requires 
\[
\bar{\mathbf{G}}\,\mathbb{J}\,\bar{\mathbf{G}}^{-1} = \mathbb{J}
\qquad\implies\qquad
[\bar{\mathbf{G}}, \mathbb{J}] = \mathbf{0}.
\]
% \end{proof}
%
The set of all matrices satisfying \eqref{eq:commutator-condition} comprise the equilibrium invariance group $G$:
\[
G \coloneq \bigl\{\,\mathbf{K} \in \mathrm{GL}(6) \;\big|\; [\mathbf{K},\mathbb{J}] = 0\,\bigr\}.
\]
This restricts the space of Class~II trace matrices \(\TraceDeDpRoot\) to a nonlinear sub-manifold, for which a simple parameterization like that of \cref{tab:trace-consts} does not exist. 
Nonetheless, any Class~II trace may be represented by the parameters of \cref{tab:trace-consts} provided \eqref{eq:commutator-condition} is satisfied. 
It is possible to conveniently parameterize a subset of Class~II traces, by first identifying a reference trace \(\mathcal{T}_R\) in the class. 
The Class~III trace developed in the following section provides such a reference. 


\subsection{Class III: Variational Consistency}\label{sec:class-iii}
A final restriction is now considered which eliminates all remaining free parameters, and furnishes a unique trace matrix called the \emph{energetic} trace. 

\begin{definition}[Class III]\label{def:class-iii}
A Class I trace is a \textbf{Class III} trace if the conjugate resultants \(\TraceTotalEnergyStress\) are exactly the classical rigid-section resultants \(\snm\):
\begin{equation}
    \TraceStress \equiv \snm .
    \label{eq:def-s3}
\end{equation}
\end{definition}

% \begin{proposition}[Class III]
All Class III traces are equivalent in the Saint~Venant class, with trace matrix determined by the consistency condition:
\begin{equation}
\TraceDeDpRoot = \IesanStiff^{-\top} \IesanEnergy
\label{eq:T0-formula}
\end{equation}
where \(\IesanEnergy\) is furnished by \eqref{eq:def-iesan-energy}.
% \end{proposition}
% \begin{proof}
The demonstration is as follows. 
For a Class I trace with $\TraceStrain(x)$ following \eqref{eq:def-T0}, 
the conjugate stiffness \eqref{eq:energy-conjugacy-def} is:
\begin{equation*}
\delta\bm{p}\cdot \left(\mathbf{1} + x\TraceEvolveEP^\top\right) \TraceDeDpRoot^\top \bm{s}(x) 
= \delta\bm{p}\cdot \left(\mathbf{1} + x\TraceEvolveEP^\top\right) \IesanEnergy\, \left(\mathbf{1} + x\TraceEvolveEP\right) \bm{p}.
\end{equation*}
% with \(\TraceEvolveEP \coloneq\TraceDpDeRoot\TraceDeDpOne\). 
Since this must hold for all $\IesanVector$, the class requires:
\begin{equation*}
\TraceDeDpRoot^\top \IesanStiff(1+x\IesanEvolveSP) = \IesanEnergy\, \left(\mathbf{1} + x\TraceEvolveEP\right).
% \label{eq:conjugacy-condition}
\end{equation*} 
with \(\mathbf{R}(x)=\IesanStiff(1+x\IesanEvolveSP).\)
Expanding both sides using $\left(\mathbf{1} + x\TraceEvolveEP\right) = \mathbf{1} + x\IesanEvolveSP$ and 
matching powers of $x$ produces:
\begin{align*}
x^0: \qquad & \TraceDeDpRoot^\top \IesanStiff = \IesanEnergy, 
\\ %\label{eq:x0-condition} \\[4pt]
x^1: \qquad & \TraceDeDpRoot^\top 
\TraceStiffNM\TraceDeDpOne% \mathbf{R}_1
= \IesanEnergy\,\IesanEvolveSP.
% \label{eq:x1-condition}
\end{align*}
Consistency in \(x^0\) furnishes the identification of \(\TraceDeDpRoot\):
\begin{equation*}
\TraceDeDpRoot = \IesanStiff^{-\top} \IesanEnergy
% \label{eq:T0-formula}
\end{equation*}
The linear evolution \(x^1\) yields the consistency condition:
\begin{equation}
\IesanEnergy\,\IesanStiff^{-1} 
\TraceStiffNM\TraceDeDpOne% \mathbf{R}_1 
= \IesanEnergy\,\IesanEvolveSP
\qquad \Longrightarrow \qquad 
\IesanStiff^{-1} 
\TraceStiffNM\TraceDeDpOne%\mathbf{R}_1
= \IesanEvolveSP
\label{eq:consistency-condition}
\end{equation}
% \end{proof}

Explicit expressions for the parameters (\cref{tab:trace-consts}) identifying the Class~III trace matrix \(\TraceDpDeRoot\) for a given cross section and material composition are derived in \cref{sec:trace-energy} and collected in \eqref{eq:energetic-trace-blocks}.
