
\subsection{Summary}

\begin{ambox}[Structural Model]{box:model}
\begin{equation}
\begin{aligned}
\bm u(x)&= 
  x\,a_\varepsilon\,\FrameBasis 
 -\tfrac{1}{2} x^2\,\IesanLoadM
 \;-\;\tfrac16 x^3\,\bm{b}_{\Section},\\[2pt]
\bm\theta(x)&= 
  x\,(a_\vartheta+c_\vartheta)\,\FrameBasis
  -\,x\,(\FrameBasis\times\IesanLoadM)\;-\;\tfrac{1}{2} x^2\,(\FrameBasis\times\bm{b}_{\Section})\;-\;\FrameBasis\times\bm{\StrainGauge}_{\gamma}.
\end{aligned}
% \label{eq:sv-kinematics-ii}
\end{equation}
The alignment deformations are \(\bm{\gamma}=\bm u'+\FrameBasis\times\bm\theta\) and \(\bm{\kappa}=\bm\theta'\):
\[
\begin{aligned}
\bm{\gamma}(x)&= a_\varepsilon\,\FrameBasis +  \bm{\StrainGauge}_{\gamma},\\[2pt]
\bm{\kappa}(x)&= 
(a_\vartheta+c_\vartheta)\,\FrameBasis
-\,(\FrameBasis\times\IesanLoadM)\;-\;x\,(\FrameBasis\times\bm{b}_{\Section}).
\end{aligned}
\]
\begin{subequations}
\begin{alignat}{8}
&\text{Transverse:}\notag\\
&&&\bm{W}_{(\varepsilon)}:=-\,\nu\,\bm{r}&&\text{ Poisson extension }& \text{1 amplitude }&&\alpha_{\varepsilon}\propto a_\varepsilon \sim x^0\\
&&&\bm{W}_{(a)}:=-\,\nu\,\operatorname{dev}\bm{r}\otimes\bm{r}&&\text{ Poisson bending }&\text{2 amplitudes }&&\bm{\alpha}_a\propto\IesanLoadM\sim x^0 \\
&&&\bm{\Pi}_{(b)}:=-\,\nu\big(\operatorname{dev}\bm{r}\otimes\bm{r}-\bm{r}\otimes\CentroidLocation\big)&&\text{ Poisson flexure }&\text{2 amplitudes }&&\bm{\alpha}_{b}\propto\bm{b}_{\Section}\sim x^1
\\
%
&\text{Longitudinal:} \notag\\
&&&\FrameBasis\otimes\bm{\WarpShearIesan}&&\text{ Flexure }&\text{2 amplitudes }&&\bm{\alpha}_{\WarpShearIesan}\propto\bm{b}_{\Section}\sim x^0 \\
&&&\FrameBasis\,\WarpTwistIesan &&\text{ Torsion }&\text{1 amplitude }&&\alpha_{\WarpTwistIesan} \propto a_{\vartheta}\sim x^0 \\
\end{alignat}
\end{subequations}
\end{ambox}
