
$$
\begin{pmatrix}
\bm{\gamma}^{\rm sh} \\
\varkappa^{\rm tw}
\end{pmatrix}
=\begin{bmatrix}
\mathbf{D}_{11} & \mathbf{D}_{12} \\
\mathbf g\mathbf{D}_{12} & D_{22}
\end{bmatrix}
\begin{pmatrix}
\ShearForce\\
T_{\mathrm C^{\rm sh}}
\end{pmatrix},
\qquad
\begin{aligned}
\mathbf{D}_{11}&=\frac{1}{GA}\,\RomanoChiShear,\\
T_{\mathrm C^{\rm sh}}&=GA^2K\,c_\vartheta.
\end{aligned}
$$

With \(\RomanoChiShear\) from \eqref{eq:def-romano-chi-shear}:
$$
\begin{aligned}
\bm{\gamma}^{\rm sh}
&=\frac{1}{GA}\,\RomanoChiShear\,\BishearTensorRB\,\bm{b}_{\Section}
+
\mathbf{D}_{12}\,\big(GA^2K\,c_\vartheta\big)\\
&=\frac{E}{GA}\, ( GA \BishearTensorRB^{-\top} \mathbf{H} \BishearTensorRB^{-1})\,\IntRGoRG\,\bm{b}_{\Section}
+
\mathbf{D}_{12}\,\big(GA^2K\,c_\vartheta\big)\\
&= (\BishearTensorRB^{-\top} \mathbf{H} \BishearTensorRB^{-1})\,E\IntRGoRG\,\bm{b}_{\Section}
+
\mathbf{D}_{12}\,\big(GA^2K\,c_\vartheta\big)\\
\end{aligned}
$$
Solving for $\bm{b}_{\Section}$:
\begin{equation}
\bm{b}_{\Section}
=\Big(\tfrac{1}{GA}\,\RomanoChiShear\,\BishearTensorRB\Big)^{-1}\!
\Big(\bm{\gamma}^{\rm sh}
- \mathbf{D}_{12}\,GA^2K \,c_\vartheta\Big).
\end{equation}

Pure shear (set $c_\vartheta=0$) or coincident centers $(\bm{r}_{\mathrm C^{\rm sh}}\equiv\bm{r}_{\mathrm C^{\rm tw}}\implies \mathbf{D}_{12}=\mathbf{0})$:
  $$
  \begin{aligned}
  \bm{b}_{\Section}
  &=\underbrace{\Big(\tfrac{1}{GA}\,\RomanoChiShear\,\BishearTensorRB\Big)^{-1}}_{\displaystyle \BfromBeta}
  \,\bm{\gamma}^{\rm sh}\\
  \end{aligned}
  $$



\begin{equation}
\begin{aligned}
\RomanoChiShear
&:=\frac{1}{A} \int_Q \mathbf{T}_{\rm sh}^T \mathbf{T}_{\rm sh} \mathrm{~d}A\\ 
&= A \int_{\Section} (G \mathbf{B}_{\rm sh}(\bm{r})\left(E \IntRGoRG\right)^{-1})^T G \mathbf{B}_{\rm sh}(\bm{r})\left(E \IntRGoRG\right)^{-1} \mathrm{~d}A\\ 
&= A \left(E \IntRGoRG\right)^{-T} \left(\int_{\Section} \mathbf{B}_{\rm sh}^T
G^2 \mathbf{B}_{\rm sh} \mathrm{~d}A\right) \left(E \IntRGoRG\right)^{-1}\\ 
&= GA \BishearTensorRB^{-\top} \mathbf{H} \BishearTensorRB^{-1}
\end{aligned}
