
\newglossaryentry{platform}{
    name=health monitoring platform,
    type=brace,
    description={The platform that hosts the \BRACE{} structural health monitoring framework and web application}
}

\newglossaryentry{asset}{
    name=Asset,
    type=brace,
    description={A unique physical object for which there is interest in its structural health}
}

\newglossaryentry{inventory}{
    name=inventory,
    type=brace,
    description={The collection of assets that are being supervised by the platform}
}

\newglossaryentry{event}{
    name=Event,
    type=brace,
    description={A situation that could place the health of a structure in jeopardy, and for which an evaluation is desired}
}

\newglossaryentry{evaluation}{
    name=Evaluation,
    type=brace,
    description={A report on the health of an asset, generated by predictors, in response to an event}
}

\newglossaryentry{predictor}{
    name=Predictor,
    type=brace,
    description={An analysis technique that an engineer can configure beforehand to process data when events are triggered}
}

\newglossaryentry{predictor-t1}{
    name=Type I Predictor, % Type I heirloom methods
    type=brace,
    description={A predictor that simulates estimated responses based on modeled behavior}
}

\newglossaryentry{predictor-t2}{
    name=Type II Predictor, % Type II data-driven methods.
    type=brace,
    description={A predictor that performs data-driven identification of structural properties}
}

\newglossaryentry{metric}{
    name=Metric,
    type=brace,
    description={Computed quantities that can inform judgment calls}
    % final quantities that are presented before a decision maker and serve as indicators of structural health or damage
}

\newglossaryentry{digital-twin}{
    name=digital twin,
    type=brace,
    description={A computational model of a unique physical asset that persistently reflects the asset's state, context and behavior}
}
