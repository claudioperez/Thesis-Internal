\subsection{Element Loading}

\begin{equation}
    g(\bm{u},\bm{v},\bm{q}) = \delta \bm{q} \cdot \int \ForceInterp^{\top}\mathbf{F}_{s}\,\bigl(\ForceInterp\bm{q} + \bm{s}_{p}(x,\bm{u}) - \bm{s}(\bm{e})\bigr)\, \dd x - \delta \bm{q} \cdot \bm{v} = 0
    \label{eq:force-residual}
\end{equation}

The total differential of the residual \eqref{eq:force-residual} is
\begin{equation*}
\begin{aligned}
& D_{u} \gv 
=\left(
\frac{\partial g}{\partial \bm{u}} \delta \bm{u}+\frac{\partial g}{\partial \bm{v}} \frac{\partial \bm{v}}{\partial \bm{u}} \delta \bm{u} + \frac{\partial g}{\partial \bm{q}} \frac{\partial \bm{q}}{\partial \bm{v}} \frac{\partial \bm{v}}{\partial \bm{u}}\right) \cdot \delta \bm{q}
+\frac{d \delta \bm{q}}{d u} \delta \bm{u} \, g = 0
\end{aligned}
\end{equation*}
The rightmost term is dropped taking \(g = 0\). 
Rearranging for \(\frac{\partial \bm{q}}{\partial \bm{v}}\) furnishes:
\begin{equation}
\delta \bm{q} \cdot \left(
\frac{\partial g}{\partial \bm{q}} \frac{\partial \bm{q}}{\partial \bm{v}}\frac{\partial \bm{v}}{\partial \bm{u}}
\right)\delta \bm{u}
= \delta \bm{q}\cdot \left( 
\frac{\partial \bm{v}}{\partial \bm{u}} - \frac{\partial g}{\partial \bm{u}}
\right)\delta \bm{u}
\label{eq:dqdv}
\end{equation}
where the relation \(\frac{\partial g}{\partial \bm{v}} = -\mathbf{1}\) is used. 
This is written compactly as
\begin{equation}
\delta \bm{q} \cdot 
\mathbf{F}\mathbf{K}_q \mathbf{A}
\delta \bm{u}
= \delta \bm{q}\cdot \left( 
\mathbf{A} - \mathbf{K}_{f}
\right)\delta \bm{u}
\label{eq:dqdv-2}
\end{equation}
with the definitions:
\begin{equation*}
\begin{aligned}
\mathbf{F}_q &\triangleq \frac{\partial \gv}{\partial \bm{q}} = \int \ForceInterp^{\top} \mathbf{F}_{s} \ForceInterp\, \dd x \\
\mathbf{K}_{f} &\triangleq \frac{\partial \gv}{\partial \bm{u}} =\frac{\partial }{\partial \bm{u}} \int \ForceInterp^{\top} \mathbf{F}_{s} \bm{s}_p(x,\bm{u})\,\dd x \\
\mathbf{A} & \triangleq \frac{\partial \bm{v}}{\partial \bm{u}} \\
\mathbf{K}_{q} &\triangleq \frac{\partial \bm{q}}{\partial \bm{v}} \\
\end{aligned}
\end{equation*}
Multiplying \eqref{eq:dqdv-2} by \(\bm{F}^{-1}\) furnishes:
\begin{equation*}
\delta \bm{q} \cdot \mathbf{K}_q \mathbf{A} \delta \bm{u}
=
\delta \bm{q} \cdot \bm{F}^{-\top} \left(\mathbf{A} - \mathbf{K}_f\right) \delta \bm{u}
\end{equation*}
Finally:
\begin{equation}
\mathbf{K}_u\delta\bm{u} = \mathbf{A}^{\top}\bm{F}^{-1}\left( 
\mathbf{A} - \mathbf{K}_f
\right)\delta \bm{u}
\label{eq:mf-tangent}
\end{equation}
