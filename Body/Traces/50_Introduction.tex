\section{Introduction}
% When no limit is placed on the number of warping fields \(\bm{\alpha}\) employed by a beam theory, it is possible to represent a continuum to any desired accuracy. 
% However this is impractical, and it is preferable to forgo auxiliary warping fields altogether whenever possible. 
% In the case of linear isotropy this is achieved by employing cross sectional constants like the shear correction factor. However, even in the simple case of linear isotropy, the subject of shear warping in beam theory remains a actively debated subject with unresolved ambiguities and strong disagreements \citep{dong2010much,dong2013principal,serpieri2014equivalence}. 

The previous section established a general constitutive framework for beam sections that is compatible with geometrically exact kinematics and multiaxial material response. 
The present section specializes the frameowork of \cref{sec:gensec} to the important case of uniform warping, where the cross-sectional distortion is governed by Saint~Venant-type modes and can be represented without introducing additional warping degrees of freedom. 


The objective is to obtain section-level constitutive relations that remain compatible with nonlinear beam equilibrium while reproducing the classical three-dimensional continuum response in the elastic regime.
To that end, enhanced strain modes are constructed to enrich the section update locally, preserving the global beam kinematics and balance laws while enforcing continuum-consistent shear and torsion behavior. 
The resulting family of models furnishes a practical, computationally efficient bridge between standard beam elements and three-dimensional elasticity.

Identifying appropriate shapes for the enhanced framework requires an association between the beam fields and the continuum displacement field. 
This association is not unique, and each choice implies a distinct interpretation of boundary conditions on the beam variables \(\bm{u}\) and \(\bm{\theta}\).
The present work addresses this by introducing the abstraction of a \emph{trace} and classifying the admissible associations within the family of beam theories in \cref{sec:gensec}.

\subsection{Literature}
The ambiguous nature of Saint~Venant's solutions is well known in the continuum mechanics literature (\cref{sec:sv}). 
However, we are not aware of any works that formalize this in the context of dimensionally reduced one-dimensional beam theories, or systematically investigate the extent to which beam theories may exactly represent three-dimensional solutions in the Saint Venant class. 
Rather, it is typically assumed at the outset that the one-dimensional model will only approximate a continuum solution. 
The subject is raised by \cite{simo1982bifurcation}, where a brief investigation is made in the context of planar buckling. 
However, because this investigation was limited to the plane, several salient three dimensional considerations do not arise such as work-conjugacy of resultants and handling of asymmetric sections.

\section{Preliminaries. Evolution Equations}\label{sec:uniform-evolution}

Our objective is to exactly recover Saint~Venant's solutions (\cref{sec:sv}) through an enhanced section constitutive model of the form \eqref{eq:enhan-strain} in the linearized one-dimensional setting of Box~\ref{box:basic-shear} under the following conditions:
\begin{enumerate}[label=(\roman*)]
    \item The body is prismatically composed heterogeneously of a initially isotropic material with homogeneous Poisson ratio \(\nu\). Consequently, Young's modulus \(E\) is:
    \[
    E(\bm{r}) = 2G(\bm{r})(1+\nu)
    \]
    where the shear modulus \(G(\bm{r})\) is constant in \(\reflenx\) and varies only in \(\bm{r}\in\Section\). 
    \item Terminal loading in the Saint~Venant class.
    \item Infinitesimal deformations.
\end{enumerate}
The restriction (i) limits attention to bodies where Saint~Venant's solution exactly satisfies the stress condition \eqref{eq:saint-venant-stress} (\cref{sec:iesan-poisson}):
\RepeatEquation{eq:saint-venant-stress}
This ensures that constitutive response in the elastic regime is exactly represented through the reduced relations:
\begin{equation*}
\bm{\sigma} = \mathbb{C}\bm{\varepsilon}
\qquad\text{with}\qquad 
\mathbb{C} = E(\bm{r}) \, \FrameBasis \otimes \FrameBasis + G(\bm{r}) \, \oneS
\end{equation*}
where the stress \(\StressVector\) and strain \(\StrainVector\) vectors are defined by \eqref{eq:def-stress-vector} and \eqref{eq:def-strain-vector}, respectively. 
This is a reasonable restriction for typical construction materials like steel and reinforced concrete, where Poisson's ratio is typically in the range \(0.23\)--\(0.27\).

\subsection{Resultant Evolution}
Under the stated conditions, the continuum equilibrium equations admit the representation:
\begin{subequations}
\label{eq:sv-equilibrium}
\begin{align}
\bm{\tau}'&=\mathbf{0} && \text{in }\Section,\\
\DivS\bm{\tau}+\AxialStress'&=0 && \text{in }\Section, \label{eq:sv-axial-equilibrium}\\
\bm{\tau}\cdot\bm\nu&=0 && \text{on }\SectionBoundary.
\end{align}
\end{subequations}
Thus the stress resultants \(\snm \coloneq (N,\bm{F},T,\bm{M})\) form a first-order initial value problem:
\begin{equation}
\left\{
\begin{aligned}
\snm'(x) & = \mathbb{J}\,\snm(x), \\
\snm(0) &= \bm{s}_0
\end{aligned}\right.
\label{eq:equilibrium-ode}
\end{equation}
where \(\mathbb{J}\) is the equilibrium generator:
\begin{equation}
\mathbb{J} \coloneq \begin{bmatrix}
 \mathbf{0} & \mathbf{0}\\
 -\FrameBasis^\times & \mathbf{0}
\end{bmatrix}
\label{eq:J-matrix}
\end{equation}
\begin{Details}
\[
\mathbb{J}^{\top}\mathbb{J} = \begin{bmatrix}
    \oneS & \mathbf{0} \\
    \mathbf{0} & \mathbf{0}
\end{bmatrix},
\quad\text{and}\quad 
\mathbb{J}\mathbb{J}^{\top} = \begin{bmatrix}
    \mathbf{0} & \mathbf{0} \\
    \mathbf{0} & \oneS
\end{bmatrix},
\]
\end{Details}
Integrating \eqref{eq:equilibrium-ode} in view of the nilpotency \(\mathbb{J}^2 = \mathbf{0}\) furnishes:
\begin{equation*}
\snm(x) = (\mathbf{1} + \reflenx\mathbb{J})\,\snm(0).
\end{equation*}
The resultants at \(\reflenx=0\) are parameterized by the six kinematic constants of the Saint~Venant particular solution \eqref{eq:iesan-general-solution}.
The resultants \(\snm\) at \(x=0\) thus depend linearly on the parameters \((a_{\varepsilon},\, \bm{a}_{\Section},\, a_{\vartheta},\, \bm{b}_{\Section})\):
\begin{equation}
\snm(0) = \IesanStiff\IesanVector
\label{eq:snm0-in-p}
\end{equation}
where \(\IesanVector\in\mathbb{R}^6\) collects the six kinematic  parameters of the particular solution: 
\begin{equation}
\IesanVector \coloneq (a_{\varepsilon},\, \bm{a}_{\Section},\, a_{\vartheta},\, \bm{b}_{\Section})
\label{eq:def-iesan-vector}
\end{equation}
In view of \eqref{eq:2-15} and \eqref{eq:2-18}:
\begin{equation}
\IesanStiff = \begin{bmatrix}
\AxialRigidity & \AxialCoupling^\top & 0 & \mathbf{0} \\[4pt]
\mathbf{0} & \mathbf{0} & \mathbf{0} & \ShearRigidity \\[4pt]
0 & \mathbf{0} & \TwistRigidity & \mathbf{0} \\[4pt]
-\FrameBasis\times \AxialCoupling& -\FrameBasis^\times \CoupleRigidity & \mathbf{0} & \mathbf{0}
\end{bmatrix}.
\label{eq:G0-neg-m}
\end{equation}
where the centroidal flexural rigidity \(\ShearRigidity\) is defined: 
\begin{equation}
    \ShearRigidity \coloneq \CoupleRigidity - \AxialCoupling\otimes\CentroidLocation
\label{eq:def-intRGoRG}
\end{equation}
\begin{Details}
Thus
\begin{equation}
\IesanStiff^{-1} =
\begin{bmatrix}
\displaystyle \frac{1}{EA} + \CentroidLocation\cdot (\ShearRigidity)^{-1}\,\CentroidLocation
& \mathbf{0}
& 0
& -\CentroidLocation^\top (\ShearRigidity)^{-1}\,\FrameBasis^\times
\\[6pt]
-(\ShearRigidity)^{-1}\,\CentroidLocation
& \mathbf{0}
& \mathbf{0}
& (\ShearRigidity)^{-1}\,\FrameBasis^\times
\\[6pt]
0
& \mathbf{0}
& \displaystyle \frac{1}{\TwistRigidity}
& \mathbf{0}
\\[6pt]
\mathbf{0}
& (\ShearRigidity)^{-1}
& \mathbf{0}
& \mathbf{0}
\end{bmatrix}
\label{eq:G0-inv-neg-m}
\end{equation}
\end{Details}
Further, introduce the evolution operator \(\IesanEvolveSP\):
\begin{equation}
\IesanEvolveSP
:=\IesanStiff^{-1}\mathbb{J}\IesanStiff 
=\begin{bmatrix}
\mathbf{0} & \oneS - \FrameBasis\otimes\CentroidLocation \\[4pt]
\mathbf{0} & \mathbf{0}
\end{bmatrix}
= \begin{bmatrix}
0 & \mathbf{0} & 0 & -\CentroidLocation^{\top} \\[4pt]
\mathbf{0} & \mathbf{0} & \mathbf{0} & \oneS \\[4pt]
0 & \mathbf{0} & 0 & \mathbf{0} \\[4pt]
\mathbf{0} & \mathbf{0} & \mathbf{0} & \mathbf{0} \\
\end{bmatrix}
\label{eq:A-matrix-neg}
\end{equation}

\begin{Details}
Using $\IesanEvolveSP^2=\mathbf 0$:
\begin{equation*}
\snm(x)=\IesanStiff\exp\left(x\IesanEvolveSP\right)\bm{p} 
=\IesanStiff(\mathbf 1+x\IesanEvolveSP)\bm p,
% \label{eq:snm-in-p}
\end{equation*}
\end{Details}
\begin{Details}
Note:
\begin{equation*}
\mathbb{J}\IesanStiff = 
\begin{pmatrix}
    \mathbf{0} & \mathbf{0} \\
    \mathbf{0} & -\FrameBasis\times(\ShearRigidity)
\end{pmatrix}
\end{equation*}
\end{Details}
